function DiagramCodeEditorTab() {
  const initialCode = `graph TD
    A[시작] --> B[처리]
    B --> C{조건}
    C -->|예| D[완료]
    C -->|아니오| B`

  const [code, setCode] = useState(initialCode)
  const [previewUrl, setPreviewUrl] = useState<string | null>(null)
  const [isRendering, setIsRendering] = useState(false)

  const handleCopy = async () => {
    try {
      await navigator.clipboard.writeText(code)
      toast.success('코드가 클립보드에 복사되었습니다')
    } catch (error) {
      toast.error('복사에 실패했습니다')
    }
  }

  const handleReset = () => {
    setCode(initialCode)
    setPreviewUrl(null)
    toast.success('코드가 초기화되었습니다')
  }

  const handleRender = async () => {
    if (!code.trim()) {
      toast.error('코드를 입력해주세요')
      return
    }

    setIsRendering(true)
    try {
      // Encode Mermaid code for mermaid.ink API
      const encoded = btoa(unescape(encodeURIComponent(code)))
      const url = `https://mermaid.ink/img/${encoded}`
      setPreviewUrl(url)
      toast.success('다이어그램이 렌더링되었습니다')
    } catch (error) {
      toast.error('렌더링에 실패했습니다')
      console.error('Rendering error:', error)
    } finally {
      setIsRendering(false)
    }
  }

  return (
    <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
      {/* Editor */}
      <div className="space-y-4">
        <div className="flex items-center justify-between">
          <h3 className="text-lg font-bold text-gray-900 dark:text-white">
            ✏️ Mermaid 코드 에디터
          </h3>
          <button
            onClick={handleRender}
            disabled={isRendering}
            className="px-4 py-2 bg-gradient-to-r from-banana-bright to-banana-golden text-gray-900 rounded-lg font-medium hover:shadow-lg transition-all disabled:opacity-50 disabled:cursor-not-allowed"
          >
            {isRendering ? '렌더링 중...' : '렌더링'}
          </button>
        </div>
        <textarea
          value={code}
          onChange={(e) => setCode(e.target.value)}
          className="w-full h-96 rounded-lg border-2 border-gray-300 bg-white p-4 font-mono text-sm text-gray-900 focus:ring-2 focus:ring-banana-bright focus:border-transparent resize-none"
          placeholder="Mermaid 코드를 입력하세요..."
        />
        <div className="flex gap-2">
          <button
            onClick={handleCopy}
            className="flex-1 px-4 py-2 border-2 border-banana-bright text-banana-dark dark:text-banana-bright rounded-lg hover:bg-banana-cream dark:hover:bg-banana-dark/20 transition-all font-medium"
          >
            코드 복사
          </button>
          <button
            onClick={handleReset}
            className="flex-1 px-4 py-2 border-2 border-banana-bright text-banana-dark dark:text-banana-bright rounded-lg hover:bg-banana-cream dark:hover:bg-banana-dark/20 transition-all font-medium"
          >
            초기화
          </button>
        </div>
      </div>

      {/* Preview */}
      <div className="bg-white rounded-xl p-6 border-2 border-banana-border">
        <h3 className="text-lg font-bold text-gray-900 dark:text-white mb-4">
          👁️ 실시간 미리보기
        </h3>
        <div className="bg-gray-50 rounded-lg p-8 min-h-96 flex items-center justify-center">
          {previewUrl ? (
            <img
              src={previewUrl}
              alt="Diagram preview"
              className="max-w-full h-auto"
              onError={() => {
                toast.error('다이어그램 렌더링에 실패했습니다')
                setPreviewUrl(null)
              }}
            />
          ) : (
            <p className="text-gray-500 dark:text-gray-400 text-center">
              렌더링 버튼을 클릭하여<br />다이어그램을 생성하세요
            </p>
          )}
        </div>
      </div>
    </div>
  )
}

      const themedCode = `%%{init: {'theme':'${mermaidTheme}'}}%%\n${sampleDiagram}`
      const encoded = btoa(unescape(encodeURIComponent(themedCode)))

      // Try mermaid.ink first, but we'll add Kroki.io as fallback in the rendering
      return {
        primary: `https://mermaid.ink/img/${encoded}`,
        fallback: `https://kroki.io/mermaid/svg/${encoded}`
      }
    } catch (error) {
      console.error(`Error generating preview for ${themeId}:`, error)
      return null
    }
  }

  const handleGenerateAll = () => {
    setIsGenerating(true)
    const urls: Record<string, string> = {}
    const fallbacks: Record<string, string> = {}

    themes.forEach(theme => {
      const urlObj = generateThemePreview(theme.id, theme.mermaidTheme)
      if (urlObj) {
        urls[theme.id] = urlObj.primary
        fallbacks[theme.id] = urlObj.fallback
      }
    })

    setPreviewUrls(urls)
    setFallbackUrls(fallbacks)
    setIsGenerating(false)
    toast.success('모든 테마가 생성되었습니다')
  }

  const handleDownloadSingle = async (themeName: string, themeId: string) => {
    const url = previewUrls[themeId]
    if (!url) {
      toast.error('먼저 테마를 생성해주세요')
      return
    }

    try {
      const response = await fetch(url)
      const blob = await response.blob()

      const link = document.createElement('a')
      link.href = URL.createObjectURL(blob)
      link.download = `${themeName}-theme.svg`
      link.click()

      toast.success(`${themeName} 테마가 다운로드되었습니다`)
    } catch (error) {
      console.error(`Download error for ${themeName}:`, error)
      toast.error('다운로드에 실패했습니다')
    }
  }

  const handleDownloadAll = async () => {
    if (Object.keys(previewUrls).length === 0) {
      toast.error('먼저 테마를 생성해주세요')
      return
    }

    try {
      const JSZip = (await import('jszip')).default
      const zip = new JSZip()

      // Fetch all images and add to zip
      const promises = themes.map(async (theme) => {
        const url = previewUrls[theme.id]
        if (!url) return

        try {
          const response = await fetch(url)
          const blob = await response.blob()
          zip.file(`${theme.name}-theme.svg`, blob)
        } catch (error) {
          console.error(`Failed to download ${theme.name}:`, error)
        }
      })

      await Promise.all(promises)

      // Generate zip file
      const content = await zip.generateAsync({ type: 'blob' })

      // Download
      const link = document.createElement('a')
      link.href = URL.createObjectURL(content)
      link.download = `diagram-themes-${Date.now()}.zip`
      link.click()

      toast.success('모든 테마가 다운로드되었습니다')
    } catch (error) {
      console.error('Download error:', error)
      toast.error('다운로드에 실패했습니다')
    }
  }

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between flex-wrap gap-3">
        <h3 className="text-xl font-bold text-gray-900 dark:text-white">
          🎨 테마 비교
        </h3>
        <div className="flex gap-2">
          <button
            onClick={handleGenerateAll}
            disabled={isGenerating}
            className="px-6 py-2 bg-gradient-to-r from-blue-500 to-blue-600 text-white rounded-lg font-medium hover:shadow-lg transition-all disabled:opacity-50"
          >
            {isGenerating ? '생성 중...' : '모든 테마 생성'}
          </button>
          <button
            onClick={handleDownloadAll}
            className="px-6 py-2 bg-gradient-to-r from-banana-bright to-banana-golden text-gray-900 rounded-lg font-medium hover:shadow-lg transition-all"
          >
            전체 다운로드
          </button>
        </div>
      </div>

      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {themes.map((theme) => (
          <div
            key={theme.id}
            className="bg-white rounded-xl p-4 border-2 border-banana-border"
          >
            <div className="flex items-center justify-between mb-3">
              <h4 className="font-semibold text-gray-900 dark:text-white">
                {theme.name} 테마
              </h4>
              {previewUrls[theme.id] && (
                <button
                  onClick={() => handleDownloadSingle(theme.name, theme.id)}
                  className="px-3 py-1 text-xs bg-gradient-to-r from-banana-bright to-banana-golden text-gray-900 rounded-lg font-medium hover:shadow-lg transition-all"
                >
                  다운로드
                </button>
              )}
            </div>
            <div className={`${theme.color} rounded-lg p-8 min-h-48 flex items-center justify-center border-2 border-gray-200 dark:border-gray-700`}>
              {previewUrls[theme.id] ? (
                <img
                  src={previewUrls[theme.id]}
                  alt={`${theme.name} theme preview`}
                  className="max-w-full h-auto"
                  onError={(e) => {
                    const target = e.target as HTMLImageElement
                    // Try fallback URL if available
                    if (fallbackUrls[theme.id] && target.src !== fallbackUrls[theme.id]) {
                      console.log(`Trying fallback for ${theme.name}: ${fallbackUrls[theme.id]}`)
                      target.src = fallbackUrls[theme.id]
                    } else {
                      console.error(`Both primary and fallback failed for ${theme.name}`)
                      toast.error(`${theme.name} 테마 렌더링 실패`)
                    }
                  }}
                />
              ) : (
                <p className="text-gray-500 text-center text-sm">
                  {theme.name}<br />테마 미리보기
                </p>
              )}
            </div>
          </div>
        ))}
      </div>
    </div>
  )
}

interface RoadmapItemProps {
  sprint: string
  title: string
  tasks: string[]
  status: 'completed' | 'in-progress' | 'planned'
}

function RoadmapItem({ sprint, title, tasks, status }: RoadmapItemProps) {
  const statusConfig = {
    completed: {
      bg: 'bg-green-100 dark:bg-green-900',
      text: 'text-green-800 dark:text-green-200',
      border: 'border-green-500',
      icon: '✅'
    },
    'in-progress': {
      bg: 'bg-blue-100 dark:bg-blue-900',
      text: 'text-blue-800 dark:text-blue-200',
      border: 'border-blue-500',
      icon: '🔄'
    },
    planned: {
      bg: 'bg-gray-100 dark:bg-gray-700',
      text: 'text-gray-800 dark:text-gray-200',
      border: 'border-gray-400',
      icon: '📋'
    }
  }

  const config = statusConfig[status]

  return (
    <div className={`rounded-lg border-l-4 ${config.border} bg-gray-50 p-4`}>
      <div className="flex items-center justify-between mb-3">
        <div>
          <h3 className="font-semibold text-gray-900 dark:text-white">
            {sprint}: {title}
          </h3>
        </div>
        <span className={`px-3 py-1 rounded-full text-sm font-medium ${config.bg} ${config.text}`}>
          {config.icon} {status.replace('-', ' ').toUpperCase()}
        </span>
      </div>
      <ul className="space-y-2">
        {tasks.map((task, index) => (
          <li key={index} className="flex items-start gap-2 text-sm text-gray-700 dark:text-gray-300">
            <span className="text-banana-bright mt-0.5">▸</span>
            {task}
          </li>
        ))}
      </ul>
    </div>
  )
}
