class ExperimentDatabase:
    """SQLite 기반 실험 데이터베이스 관리"""
    
    def __init__(self, db_path=DB_PATH):
        self.db_path = str(db_path)
        self.init_database()

    def get_connection(self):
        """스레드 안전 SQLite 연결 반환 (WAL 모드)"""
        conn = sqlite3.connect(self.db_path, check_same_thread=False, timeout=10)
        conn.execute("PRAGMA journal_mode=WAL")
        conn.execute("PRAGMA synchronous=NORMAL")
        return conn
    
    def init_database(self):
        """데이터베이스 테이블 초기화"""
        conn = self.get_connection()
        cursor = conn.cursor()
        
        # 실험 메타데이터 테이블
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS experiments (
                id TEXT PRIMARY KEY,
                status TEXT NOT NULL,
                method TEXT NOT NULL,
                config TEXT NOT NULL,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                started_at TIMESTAMP,
                completed_at TIMESTAMP,
                current_step INTEGER DEFAULT 0,
                total_steps INTEGER NOT NULL,
                best_accuracy REAL DEFAULT 0.0,
                final_accuracy REAL,
                results_pkl BLOB,
                error_message TEXT,
                model1_hf TEXT,
                model2_hf TEXT,
                merge_ratio REAL,
                merge_method TEXT
            )
        ''')
        
        # 실험 진행 로그 테이블
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS experiment_logs (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                experiment_id TEXT NOT NULL,
                step INTEGER NOT NULL,
                timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                train_acc REAL,
                test_acc REAL,
                fitness REAL,
                metrics TEXT,
                FOREIGN KEY (experiment_id) REFERENCES experiments(id) ON DELETE CASCADE
            )
        ''')
        
        # 계보 추적 테이블
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS lineage_records (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                experiment_id TEXT NOT NULL,
                generation INTEGER NOT NULL,
                parent1_id TEXT,
                parent2_id TEXT,
                parent1_model_path TEXT,  
                parent2_model_path TEXT,                  
                child_id TEXT NOT NULL,
                improvement_percent REAL,
                timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                FOREIGN KEY (experiment_id) REFERENCES experiments(id) ON DELETE CASCADE
            )
        ''')
        
        # 인덱스 생성 for 빠른 쿼리
        cursor.execute('CREATE INDEX IF NOT EXISTS idx_exp_status ON experiments(status)')
        cursor.execute('CREATE INDEX IF NOT EXISTS idx_exp_created ON experiments(created_at)')
        cursor.execute('CREATE INDEX IF NOT EXISTS idx_logs_exp ON experiment_logs(experiment_id)')
        cursor.execute('CREATE INDEX IF NOT EXISTS idx_lineage_exp ON lineage_records(experiment_id)')
        
        conn.commit()
        conn.close()

class ExperimentManager:
    """실험 관리 시스템"""
    
    def __init__(self):
        self.db = ExperimentDatabase()
        self.running_threads = {}
        
    def generate_experiment_id(self):
        """짧고 기억하기 쉬운 실험 ID 생성"""
        return str(uuid.uuid4())[:8]

    def create_experiment(self, config):
        """새 실험 생성 및 DB 등록"""
        exp_id = self.generate_experiment_id()
        
        # FitnessObjective를 직렬화 가능한 형태로 변환
        config_copy = config.copy()
        if 'fitness_objectives' in config_copy:
            config_copy['fitness_objectives'] = [obj.value for obj in config_copy['fitness_objectives']]
        if 'fitness_weights' in config_copy:
            config_copy['fitness_weights'] = {
                obj.value if isinstance(obj, FitnessObjective) else obj: weight 
                for obj, weight in config_copy['fitness_weights'].items()
            }
        
        conn = self.db.get_connection()
        cursor = conn.cursor()
        
        cursor.execute('''
            INSERT INTO experiments (
                id, status, method, config, total_steps,
                model1_hf, model2_hf, merge_ratio, merge_method
            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
        ''', (
            exp_id,
            'pending',
            config.get('method', 'unknown'),
            json.dumps(config_copy),  
            config.get('total_forward_passes', 0),
            config.get('model1_hf', ''),
            config.get('model2_hf', ''),
            config.get('merge_ratio', 0.5),
            config.get('merge_method', 'slerp')
        ))
        
        conn.commit()
        conn.close()
        
        return exp_id
    

    
    def start_experiment(self, exp_id, config):
        """백그라운드에서 실험 시작"""
        # 상태를 running으로 업데이트
        self.update_experiment_status(exp_id, 'running')
        
        # 백그라운드 스레드 시작
        thread = threading.Thread(
            target=self._run_experiment_background,
            args=(exp_id, config),
            daemon=True
        )
        self.running_threads[exp_id] = thread
        thread.start()
        
        return exp_id

    def _run_experiment_background(self, exp_id, config):
        """백그라운드에서 실험 실행"""
        try:
            # 시작 시간 기록
            conn = self.db.get_connection()
            cursor = conn.cursor()
            cursor.execute(
                "UPDATE experiments SET started_at = CURRENT_TIMESTAMP WHERE id = ?",
                (exp_id,)
            )
            conn.commit()
            conn.close()
            
            # fitness_objectives를 Enum으로 복원
            if 'fitness_objectives' in config:
                config['fitness_objectives'] = [
                    FitnessObjective(obj) if isinstance(obj, str) else obj 
                    for obj in config['fitness_objectives']
                ]
            
            if 'fitness_weights' in config:
                config['fitness_weights'] = {
                    FitnessObjective(k) if isinstance(k, str) else k: v 
                    for k, v in config['fitness_weights'].items()
                }
            
            # 실험 메서드에 따라 실행
            method = config.get('method', 'natural_niches')
            
            if method in ['natural_niches', 'ga']:
                results = self._run_natural_niches_with_db(exp_id, config)
            elif method == 'map_elites':
                results = self._run_map_elites_with_db(exp_id, config)
            elif method == 'cma_es':
                results = self._run_cma_es_with_db(exp_id, config)
            elif method == 'brute_force':
                results = self._run_brute_force_with_db(exp_id, config)
            elif method == 'darwin_mergekit':
                results = self._run_darwin_mergekit_with_db(exp_id, config)
            else:
                raise ValueError(f"Unknown method: {method}")
            
            # 성공적으로 완료
            self._save_final_results(exp_id, results)
            self.update_experiment_status(exp_id, 'completed')
            
        except Exception as e:
            # 에러 처리
            import traceback
            error_msg = f"{str(e)}\n{traceback.format_exc()}"
            self.update_experiment_status(exp_id, 'error', error_msg)
    

    
    def update_experiment_status(self, exp_id, status, error_msg=None):
        """실험 상태 업데이트"""
        conn = self.db.get_connection()
        cursor = conn.cursor()
        
        if status == 'completed':
            cursor.execute('''
                UPDATE experiments 
                SET status = ?, updated_at = CURRENT_TIMESTAMP, 
                    completed_at = CURRENT_TIMESTAMP
                WHERE id = ?
            ''', (status, exp_id))
        elif status == 'error':
            cursor.execute('''
                UPDATE experiments 
                SET status = ?, updated_at = CURRENT_TIMESTAMP,
                    error_message = ?
                WHERE id = ?
            ''', (status, error_msg, exp_id))
        else:
            cursor.execute('''
                UPDATE experiments 
                SET status = ?, updated_at = CURRENT_TIMESTAMP
                WHERE id = ?
            ''', (status, exp_id))
        
        conn.commit()
        conn.close()
    
    def log_progress(self, exp_id, step, train_acc, test_acc, fitness, metrics):
        """진행 상황 로깅"""
        conn = self.db.get_connection()
        cursor = conn.cursor()
        
        # 로그 추가
        cursor.execute('''
            INSERT INTO experiment_logs 
            (experiment_id, step, train_acc, test_acc, fitness, metrics)
            VALUES (?, ?, ?, ?, ?, ?)
        ''', (exp_id, step, train_acc, test_acc, fitness, json.dumps(metrics)))
        
        # 현재 스텝과 최고 정확도 업데이트
        cursor.execute('''
            UPDATE experiments 
            SET current_step = ?, 
                best_accuracy = MAX(best_accuracy, ?),
                updated_at = CURRENT_TIMESTAMP
            WHERE id = ?
        ''', (step, test_acc, exp_id))
        
        conn.commit()
        conn.close()

    def log_lineage(self, exp_id, generation, parent1_id, parent2_id,

        """계보 정보 로깅"""

        cursor.execute('''
            INSERT INTO lineage_records 
            (experiment_id, generation, parent1_id, parent2_id,

        ''', (exp_id, generation, parent1_id, parent2_id, 
              parent1_model_path, parent2_model_path, child_id, improvement)) 
        
        conn.commit()
        conn.close()

        merge_method = config.get('merge_method', 'slerp')
        
        # 피트니스 설정
        fitness_objectives = config.get('fitness_objectives', [FitnessObjective.ACCURACY])
        fitness_weights = config.get('fitness_weights', {FitnessObjective.ACCURACY: 1.0})
        track_lineage = config.get('track_lineage', True)
        
        # 데이터 로드
        (x_train, y_train), (x_test, y_test) = load_data()
        
        use_matchmaker = not no_matchmaker
        use_crossover = not no_crossover
        use_splitpoint = not no_splitpoint
        
        # Initialize adaptive fitness evaluator
        fitness_evaluator = AdaptiveFitnessEvaluator(fitness_objectives, fitness_weights)
        
        # Initialize lineage tracker
        lineage_tracker = LineageTracker() if track_lineage else None
        
        results = []
        
        if use_pre_trained:
            model_1, model_2 = get_pre_trained_models(
                model1_hf, model2_hf, model1_file, model2_file, merge_ratio, merge_method
            )


        parent1_path = config.get('model1_hf', '') or str(config.get('model1_file', ''))
        parent2_path = config.get('model2_hf', '') or str(config.get('model2_file', ''))


        
        for run in range(runs):
            result = defaultdict(list)
            seed = 42 + run
            
            # Initialize adaptive parameters
            std_base = 0.01
            std_current = std_base
            improvement_window = deque(maxlen=50)
            
            if JAX_AVAILABLE:
                key = jax.random.PRNGKey(seed)
                archive = jnp.zeros([pop_size, num_params])
                scores = jnp.zeros([pop_size, len(x_train)], dtype=jnp.bool_)
            else:
                np.random.seed(seed)
                archive = np.zeros([pop_size, num_params])
                scores = np.zeros([pop_size, len(x_train)], dtype=bool)
            
            if not use_pre_trained:
                if JAX_AVAILABLE:
                    key, key1, key2 = jax.random.split(key, 3)
                    model_1 = jax.random.normal(key1, (num_params,)) * 0.01
                    model_2 = jax.random.normal(key2, (num_params,)) * 0.01
                else:
                    model_1 = np.random.randn(num_params) * 0.01
                    model_2 = np.random.randn(num_params) * 0.01
            
            # Initialize archive with two models
            for model in (model_1, model_2):
                logits = mlp(model, x_train)
                if JAX_AVAILABLE:
                    score = jnp.argmax(logits, axis=1) == y_train
                else:
                    score = np.argmax(logits, axis=1) == y_train
                archive, scores = update_archive(score, model, archive, scores, alpha)
                
                if lineage_tracker:
                    model_id = lineage_tracker.get_model_id(model)
                    metrics = comprehensive_evaluation(model, x_train, y_train, x_test, y_test)
                    lineage_tracker.record_performance(model_id, metrics)
            
            # Evolution loop with DB logging
            for i in range(total_forward_passes):
                progress = i / total_forward_passes
                
                # Update fitness weights dynamically
                fitness_evaluator.update_weights_dynamically(i, total_forward_passes)
                
                # Update adaptive mutation std
                if i > 50 and i % 25 == 0:
                    if len(improvement_window) >= 25:
                        recent_improvements = list(improvement_window)[-25:]
                        avg_improvement = np.mean(recent_improvements)
                        
                        if avg_improvement < 0.001:  # Stagnation
                            std_current = min(std_current * 1.2, 0.1)
                        else:  # Good progress
                            std_current = max(std_current * 0.95, 0.002)
                
                # Determine novelty weight based on progress
                if progress < 0.5:
                    novelty_lambda = 0.0
                elif progress < 0.7:
                    novelty_lambda = 0.2
                else:
                    novelty_lambda = 0.4
                
                if JAX_AVAILABLE:
                    k1, k2, k3, key = jax.random.split(key, 4)
                else:
                    k1 = k2 = k3 = None
                
                # Sample parents with novelty weighting
                parent1, parent2, p1_idx, p2_idx = sample_parents_with_novelty(
                    archive, scores, k1, alpha, use_matchmaker, novelty_lambda
                )
                
                # Check parent diversity and resample if too similar
                if progress > 0.5:
                    parent_diversity = compute_cosine_diversity(parent1, parent2)
                    if parent_diversity < 0.1:
                        if JAX_AVAILABLE:
                            k_resample, key = jax.random.split(key)
                            p2_idx = jax.random.randint(k_resample, shape=(), 
                                                       minval=0, maxval=pop_size)
                        else:
                            p2_idx = np.random.randint(0, pop_size)
                        parent2 = archive[p2_idx]
                
                # Create child with advanced merging in late stage
                if use_crossover:
                    if progress > 0.7 and np.random.random() < 0.3:
                        child = enhanced_model_manager.merge_models_advanced(
                            parent1, parent2, method="fisher_weighted", 
                            ratio=np.random.uniform(0.3, 0.7)
                        )
                    else:
                        if use_splitpoint:
                            child = crossover([parent1, parent2], k2)
                        else:
                            child = crossover_without_splitpoint([parent1, parent2], k2)
                else:
                    child = parent1.copy()
                
                # Apply adaptive mutation
                if not use_pre_trained:
                    child = mutate(child, k3, std=std_current)
                
                # Apply Lamarckian refinement in late stage
                if progress > 0.7 and len(result["test_values"]) > 100:
                    recent_fitness = result["multi_objective_fitness"][-100:] if "multi_objective_fitness" in result else []
                    if recent_fitness:
                        threshold = np.percentile(recent_fitness, 80)
                        child_fitness = fitness_evaluator.evaluate(child, x_train, y_train, x_test, y_test)
                        
                        if child_fitness > threshold:
                            child = apply_lamarckian_refinement(
                                child, x_train, y_train, 
                                subset_size=256, lr=0.001, epochs=2
                            )
                
                # Track lineage
                if lineage_tracker:
                    child_id = lineage_tracker.record_birth(child, parent1, parent2, i)
                    child_metrics = comprehensive_evaluation(child, x_train, y_train, x_test, y_test)
                    lineage_tracker.record_performance(child_id, child_metrics)
                    improvements = lineage_tracker.evaluate_improvement(child_id)
                    
                    # DB에 계보 정보 저장
                    if improvements and 'test_acc' in improvements:

                        self.log_lineage(
                            exp_id, i,
                            str(p1_idx), str(p2_idx),
                            str(child_id),
                            improvements['test_acc'],
                            parent1_model_path=parent1_path,  # 키워드 인자로 변경
                            parent2_model_path=parent2_path   # 키워드 인자로 변경
                        )


                
                # Use adaptive fitness evaluation
                fitness = fitness_evaluator.evaluate(child, x_train, y_train, x_test, y_test)
                
                logits = mlp(child, x_train)
                if JAX_AVAILABLE:
                    score = jnp.argmax(logits, axis=1) == y_train
                else:
                    score = np.argmax(logits, axis=1) == y_train
                archive, scores = update_archive(score, child, archive, scores, alpha)
                
                # Log results
                result["evals"].append(i)
                result["multi_objective_fitness"].append(fitness)
                train_acc = scores.mean(axis=1)
                
                if store_train_results:
                    result["train_values"].append(float(train_acc.max()))
                
                # Test accuracy
                best_individual = train_acc.argmax()
                logits = mlp(archive[best_individual], x_test)
                acc = get_acc(logits, y_test)
                result["test_values"].append(float(acc))
                
                # Track improvement for adaptive std
                if len(result["test_values"]) > 1:
                    improvement = result["test_values"][-1] - result["test_values"][-2]
                    improvement_window.append(improvement)
                
                # Store additional metrics
                for obj in fitness_objectives:
                    if obj in fitness_evaluator.history:
                        result[f"fitness_{obj.value}"].append(fitness_evaluator.history[obj][-1])
                
                # DB에 진행상황 로깅 (100 스텝마다)
                if i % 100 == 0:
                    # 전체 진행률을 올바르게 계산
                    total_steps_done = run * total_forward_passes + i
                    total_steps_all = runs * total_forward_passes

                    self.log_progress(
                        exp_id,
                        total_steps_done,  
                        float(train_acc.max()),
                        float(acc),
                        float(fitness),
                        {
                            'run': run,
                            'std': float(std_current),
                            'novelty_lambda': float(novelty_lambda),
                            'total_progress': total_steps_done / total_steps_all
                        }
                    )


            
            # Find Pareto front at the end
            pareto_indices = fitness_evaluator.get_pareto_front(archive, x_train, y_train)
            result["pareto_front_size"] = len(pareto_indices)
            
            # Add lineage validation results
            if lineage_tracker:
                result["lineage_validation"] = lineage_tracker.get_statistical_validation()
                result["total_improvements"] = len([imp for imp in lineage_tracker.improvement_history 
                                                   if imp['improvements'].get('test_acc', 0) > 0])
            
            results.append(dict(result))
        
        return results

    def export_experiment(self, exp_id, format='pkl'):

        if format == 'pkl':

                filename = f"logs_{exp_id}_{datetime.now().strftime('%Y%m%d_%H%M%S')}.csv"
                temp_filepath = os.path.join(tempfile.gettempdir(), filename)
                
                logs.to_csv(temp_filepath, index=False)
                
                return temp_filepath
        
        return None

def compute_cosine_diversity(a, b):
    """Compute cosine distance between two parameter vectors"""
    eps = 1e-8
    dot_product = np.dot(a, b)
    norm_a = np.linalg.norm(a) + eps
    norm_b = np.linalg.norm(b) + eps
    similarity = dot_product / (norm_a * norm_b)
    return 1.0 - similarity

def get_adaptive_mutation_std(improvement_history, base_std=0.01, 
                            stagnation_threshold=0.001,
                            increase_factor=1.2, decrease_factor=0.95):
    """Compute adaptive mutation standard deviation based on recent improvements"""
    if len(improvement_history) < 10:
        return base_std
    
    recent_improvements = improvement_history[-10:]
    avg_improvement = np.mean(recent_improvements)
    
    if avg_improvement < stagnation_threshold:
        # Stagnation detected - increase exploration
        new_std = min(base_std * increase_factor, 0.1)
    else:
        # Good progress - refine search
        new_std = max(base_std * decrease_factor, 0.002)
    
    return new_std

def apply_lamarckian_refinement(params, x_train, y_train, 
                               subset_size=256, lr=0.001, epochs=2):
    """Apply small-scale fine-tuning to promising individuals"""
    # Sample a subset for efficiency
    if len(x_train) > subset_size:
        idx = np.random.choice(len(x_train), subset_size, replace=False)
        x_subset = x_train[idx]
        y_subset = y_train[idx]
    else:
        x_subset = x_train
        y_subset = y_train
    
    # Apply lightweight training
    refined_params = train(params, x_subset, y_subset, lr=lr, num_epochs=epochs)
    return refined_params

class LineageTracker:
    """부모-자식 계보 추적 및 성능 검증 시스템"""
    
    def __init__(self):
        self.genealogy = {}  # {child_id: (parent1_id, parent2_id)}
        self.performance_history = {}  # {model_id: metrics}
        self.generation_map = {}  # {generation: [model_ids]}
        self.improvement_history = []  # List of improvement records
        
    def get_model_id(self, params):
        """Generate unique ID for model parameters"""
        if JAX_AVAILABLE and hasattr(params, 'tobytes'):
            return hash(params.tobytes())
        else:
            return hash(params.tobytes() if hasattr(params, 'tobytes') else str(params))
    
    def record_birth(self, child_params, parent1_params, parent2_params, generation):
        """자식 생성 기록"""
        child_id = self.get_model_id(child_params)
        parent1_id = self.get_model_id(parent1_params)
        parent2_id = self.get_model_id(parent2_params)
        
        self.genealogy[child_id] = (parent1_id, parent2_id)
        
        if generation not in self.generation_map:
            self.generation_map[generation] = []
        self.generation_map[generation].append(child_id)
        
        return child_id
    
    def record_performance(self, model_id, metrics):
        """Record performance metrics for a model"""
        self.performance_history[model_id] = metrics
    
    def evaluate_improvement(self, child_id):
        """부모 대비 개선도 계산"""
        if child_id not in self.genealogy:
            return None
            
        parent1_id, parent2_id = self.genealogy[child_id]
        
        child_perf = self.performance_history.get(child_id, {})
        parent1_perf = self.performance_history.get(parent1_id, {})
        parent2_perf = self.performance_history.get(parent2_id, {})
        
        if not (child_perf and parent1_perf and parent2_perf):
            return None
        
        # 부모 평균 대비 개선율
        improvements = {}
        for metric in child_perf:
            if metric in parent1_perf and metric in parent2_perf:
                parent_avg = (parent1_perf[metric] + parent2_perf[metric]) / 2
                if parent_avg > 0:
                    improvements[metric] = ((child_perf[metric] - parent_avg) / parent_avg) * 100
                else:
                    improvements[metric] = 0
        
        # Record improvement
        improvement_record = {
            'child_id': child_id,
            'parent1_id': parent1_id,
            'parent2_id': parent2_id,
            'improvements': improvements,
            'child_metrics': child_perf,
            'parent1_metrics': parent1_perf,
            'parent2_metrics': parent2_perf
        }
        self.improvement_history.append(improvement_record)
        
        return improvements
    
    def get_statistical_validation(self, num_samples=100):
        """통계적 유의성 검증"""
        if not SCIPY_AVAILABLE:
            return {"error": "SciPy not available for statistical tests"}
        
        if len(self.improvement_history) < 2:
            return {"error": "Not enough data for statistical analysis"}
        
        # Collect parent and child performances
        parent_scores = []
        child_scores = []
        
        for record in self.improvement_history[-num_samples:]:
            if 'test_acc' in record['child_metrics']:
                parent_avg = (record['parent1_metrics'].get('test_acc', 0) + 
                            record['parent2_metrics'].get('test_acc', 0)) / 2
                parent_scores.append(parent_avg)
                child_scores.append(record['child_metrics']['test_acc'])
        
        if len(parent_scores) < 2:
            return {"error": "Not enough test accuracy data"}
        
        # Perform paired t-test
        t_stat, p_value = stats.ttest_rel(child_scores, parent_scores)
        
        # Calculate effect size (Cohen's d)
        diff = np.array(child_scores) - np.array(parent_scores)
        cohen_d = np.mean(diff) / np.std(diff) if np.std(diff) > 0 else 0
        
        return {
            'mean_parent': np.mean(parent_scores),
            'mean_child': np.mean(child_scores),
            'improvement': np.mean(child_scores) - np.mean(parent_scores),
            'improvement_percent': ((np.mean(child_scores) - np.mean(parent_scores)) / 
                                   np.mean(parent_scores) * 100) if np.mean(parent_scores) > 0 else 0,
            't_statistic': t_stat,
            'p_value': p_value,
            'significant': p_value < 0.05,
            'cohen_d': cohen_d,
            'sample_size': len(parent_scores)
        }

def comprehensive_evaluation(model_params, x_train, y_train, x_test, y_test):
    """포괄적 성능 평가 with enhanced metrics"""
    metrics = {}
    
    # 1. 기본 정확도
    train_logits = mlp(model_params, x_train)
    test_logits = mlp(model_params, x_test)
    metrics['train_acc'] = float(get_acc(train_logits, y_train))
    metrics['test_acc'] = float(get_acc(test_logits, y_test))
    
    # 2. 일반화 갭 (낮을수록 좋음)
    metrics['generalization_gap'] = abs(metrics['train_acc'] - metrics['test_acc'])
    
    # 3. 손실값
    metrics['train_loss'] = float(get_loss(model_params, x_train, y_train))
    metrics['test_loss'] = float(get_loss(model_params, x_test, y_test))
    
    # 4. 파라미터 효율성 (sparsity)
    metrics['sparsity'] = float(np.sum(np.abs(model_params) < 0.01) / len(model_params))
    
    # 5. 노이즈 강건성
    noise_levels = [0.05, 0.1, 0.2]
    robustness_scores = []
    for noise in noise_levels:
        noisy_x = x_test + np.random.randn(*x_test.shape) * noise
        noisy_logits = mlp(model_params, noisy_x)
        robustness_scores.append(float(get_acc(noisy_logits, y_test)))
    metrics['robustness'] = float(np.mean(robustness_scores))
    metrics['robustness_decline'] = metrics['test_acc'] - metrics['robustness']
    
    # 6. 추론 속도
    start_time = time.perf_counter()
    for _ in range(100):
        _ = mlp(model_params, x_test[:32] if len(x_test) > 32 else x_test)
    metrics['inference_time'] = (time.perf_counter() - start_time) / 100
    
    # 7. Confidence metrics
    if JAX_AVAILABLE:
        probs = jax.nn.softmax(test_logits)
        max_probs = jnp.max(probs, axis=1)
        metrics['avg_confidence'] = float(jnp.mean(max_probs))
        metrics['confidence_std'] = float(jnp.std(max_probs))
    else:
        exp_logits = np.exp(test_logits - np.max(test_logits, axis=1, keepdims=True))
        probs = exp_logits / np.sum(exp_logits, axis=1, keepdims=True)
        max_probs = np.max(probs, axis=1)
        metrics['avg_confidence'] = float(np.mean(max_probs))
        metrics['confidence_std'] = float(np.std(max_probs))
    
    # 8. Parameter magnitude statistics
    metrics['param_mean'] = float(np.mean(np.abs(model_params)))
    metrics['param_std'] = float(np.std(model_params))
    metrics['param_max'] = float(np.max(np.abs(model_params)))
    
    return metrics

@dataclass
class LayerMapping:
    """Represents a mapping between layers of two models"""
    source_layer: str
    target_layer: str
    transform_matrix: Optional[np.ndarray] = None
    compatibility_score: float = 0.0

class ArchitectureMapper:
    """Intelligent architecture matching and alignment system"""
    
    def __init__(self):
        self.layer_mappings = []
        self.alignment_cache = {}
        
    def analyze_architecture(self, model) -> Dict:
        """Analyze model architecture and extract layer information"""
        architecture_info = {
            'layers': {},
            'total_params': 0,
            'layer_types': defaultdict(int)
        }
        
        if model is None:
            return architecture_info
            
        if hasattr(model, 'named_parameters'):
            for name, param in model.named_parameters():
                layer_type = self._identify_layer_type(name)
                architecture_info['layers'][name] = {
                    'shape': tuple(param.shape),
                    'type': layer_type,
                    'params': param.numel()
                }
                architecture_info['layer_types'][layer_type] += 1
                architecture_info['total_params'] += param.numel()
                
        return architecture_info
    
    def _identify_layer_type(self, layer_name: str) -> str:
        """Identify the type of layer from its name"""
        layer_name_lower = layer_name.lower()
        if 'conv' in layer_name_lower:
            return 'conv'
        elif 'linear' in layer_name_lower or 'fc' in layer_name_lower or 'dense' in layer_name_lower:
            return 'linear'
        elif 'norm' in layer_name_lower:
            return 'norm'
        elif 'attention' in layer_name_lower:
            return 'attention'
        elif 'embedding' in layer_name_lower:
            return 'embedding'
        else:
            return 'other'
    
    def compute_layer_compatibility(self, layer1_info: Dict, layer2_info: Dict) -> float:
        """Compute compatibility score between two layers"""
        # Type compatibility
        if layer1_info['type'] != layer2_info['type']:
            type_score = 0.5
        else:
            type_score = 1.0
        
        # Shape compatibility
        shape1, shape2 = layer1_info['shape'], layer2_info['shape']
        
        if len(shape1) != len(shape2):
            dim_score = 0.3
        else:
            dim_diffs = [abs(s1 - s2) / max(s1, s2) for s1, s2 in zip(shape1, shape2)]
            dim_score = 1.0 - np.mean(dim_diffs)
        
        # Parameter count compatibility
        param_ratio = min(layer1_info['params'], layer2_info['params']) / max(layer1_info['params'], layer2_info['params'])
        
        # Weighted compatibility score
        compatibility = 0.4 * type_score + 0.4 * dim_score + 0.2 * param_ratio
        
        return compatibility
    
    def find_optimal_mapping(self, arch1: Dict, arch2: Dict) -> List[LayerMapping]:
        """Find optimal layer mappings between two architectures"""
        mappings = []
        
        layers1 = list(arch1['layers'].keys())
        layers2 = list(arch2['layers'].keys())
        
        # Create compatibility matrix
        compatibility_matrix = np.zeros((len(layers1), len(layers2)))
        
        for i, l1 in enumerate(layers1):
            for j, l2 in enumerate(layers2):
                compatibility_matrix[i, j] = self.compute_layer_compatibility(
                    arch1['layers'][l1], arch2['layers'][l2]
                )
        
        # Hungarian algorithm for optimal matching (simplified greedy version)
        used_targets = set()
        
        for i in range(len(layers1)):
            if i >= len(layers2):
                break
                
            # Find best match for this source layer
            best_j = -1
            best_score = -1
            
            for j in range(len(layers2)):
                if j not in used_targets and compatibility_matrix[i, j] > best_score:
                    best_j = j
                    best_score = compatibility_matrix[i, j]
            
            if best_j >= 0 and best_score > 0.3:  # Minimum compatibility threshold
                used_targets.add(best_j)
                
                # Create transformation matrix if needed
                transform = self._create_transform_matrix(
                    arch1['layers'][layers1[i]]['shape'],
                    arch2['layers'][layers2[best_j]]['shape']
                )
                
                mappings.append(LayerMapping(
                    source_layer=layers1[i],
                    target_layer=layers2[best_j],
                    transform_matrix=transform,
                    compatibility_score=best_score
                ))
        
        return mappings
    
    def _create_transform_matrix(self, shape1: Tuple, shape2: Tuple) -> Optional[np.ndarray]:
        """Create transformation matrix for incompatible shapes"""
        if shape1 == shape2:
            return None
        
        # Simple linear transformation for 2D weight matrices
        if len(shape1) == 2 and len(shape2) == 2:
            transform = np.random.randn(shape2[0], shape1[0]) * 0.01
            return transform
        
        return None
    
    def align_parameters(self, params1: np.ndarray, params2: np.ndarray, 
                         model1=None, model2=None) -> Tuple[np.ndarray, np.ndarray]:
        """Align parameters from two models for effective merging"""
        
        # If we have actual models, use intelligent alignment
        if model1 is not None and model2 is not None:
            arch1 = self.analyze_architecture(model1)
            arch2 = self.analyze_architecture(model2)
            
            mappings = self.find_optimal_mapping(arch1, arch2)
            
            avg_compatibility = np.mean([m.compatibility_score for m in mappings]) if mappings else 0
            
            print(f"Architecture alignment score: {avg_compatibility:.3f}")
            
            # If compatibility is too low, use fallback method
            if avg_compatibility < 0.5:
                return self._fallback_alignment(params1, params2)
        
        # Ensure same dimensions
        target_len = max(len(params1), len(params2))
        
        if len(params1) < target_len:
            params1 = np.pad(params1, (0, target_len - len(params1)), 'constant')
        else:
            params1 = params1[:target_len]
            
        if len(params2) < target_len:
            params2 = np.pad(params2, (0, target_len - len(params2)), 'constant')
        else:
            params2 = params2[:target_len]
        
        return params1, params2
    
    def _fallback_alignment(self, params1: np.ndarray, params2: np.ndarray) -> Tuple[np.ndarray, np.ndarray]:
        """Fallback alignment method when automatic matching fails"""
        min_len = min(len(params1), len(params2))
        
        # Find best alignment offset using correlation
        best_offset = 0
        best_corr = -1
        
        for offset in range(0, min(100, min_len // 10)):
            if offset + min_len <= len(params1) and offset + min_len <= len(params2):
                corr = np.corrcoef(params1[offset:offset+min_len], 
                                  params2[offset:offset+min_len])[0, 1]
                if corr > best_corr:
                    best_corr = corr
                    best_offset = offset
        
        # Apply best offset
        if best_offset > 0:
            params1 = params1[best_offset:]
            params2 = params2[best_offset:]
        
        return params1, params2

class FitnessObjective(Enum):
    """Available fitness objectives"""
    ACCURACY = "accuracy"
    SPEED = "speed"
    SIZE = "size"
    ROBUSTNESS = "robustness"
    GENERALIZATION = "generalization"

class AdaptiveFitnessEvaluator:
    """Multi-objective adaptive fitness evaluation with enhanced late-stage weights"""
    
    def __init__(self, objectives: List[FitnessObjective] = None, 
                 weights: Dict[FitnessObjective, float] = None):
        self.objectives = objectives or [FitnessObjective.ACCURACY]
        self.weights = weights or {FitnessObjective.ACCURACY: 1.0}
        self.history = defaultdict(list)
        self.baseline_metrics = {}
        
    def evaluate(self, params, x_train, y_train, x_test=None, y_test=None) -> float:
        """Evaluate fitness with multiple objectives"""
        scores = {}
        
        # Accuracy
        if FitnessObjective.ACCURACY in self.objectives:
            logits = mlp(params, x_train)
            scores[FitnessObjective.ACCURACY] = float(get_acc(logits, y_train))
        
        # Speed (inverse of inference time)
        if FitnessObjective.SPEED in self.objectives:
            start_time = time.perf_counter()
            for _ in range(10):
                _ = mlp(params, x_train[:100])
            inference_time = (time.perf_counter() - start_time) / 10
            scores[FitnessObjective.SPEED] = 1.0 / (1.0 + inference_time)
        
        # Size (inverse of parameter count)
        if FitnessObjective.SIZE in self.objectives:
            sparsity = np.sum(np.abs(params) < 0.01) / len(params)
            scores[FitnessObjective.SIZE] = sparsity
        
        # Robustness (performance on noisy data)
        if FitnessObjective.ROBUSTNESS in self.objectives:
            noise_level = 0.1
            x_noisy = x_train + np.random.randn(*x_train.shape) * noise_level
            logits_noisy = mlp(params, x_noisy)
            scores[FitnessObjective.ROBUSTNESS] = float(get_acc(logits_noisy, y_train))
        
        # Generalization (if test data provided)
        if FitnessObjective.GENERALIZATION in self.objectives and x_test is not None:
            logits_test = mlp(params, x_test)
            test_acc = float(get_acc(logits_test, y_test))
            train_acc = scores.get(FitnessObjective.ACCURACY, 0)
            scores[FitnessObjective.GENERALIZATION] = 1.0 - abs(train_acc - test_acc)
        
        # Weighted combination
        total_fitness = 0
        total_weight = 0
        
        for obj, score in scores.items():
            weight = self.weights.get(obj, 1.0)
            total_fitness += score * weight
            total_weight += weight
            
            # Track history
            self.history[obj].append(score)
        
        if total_weight > 0:
            total_fitness /= total_weight
            
        return total_fitness
    
    def update_weights_dynamically(self, generation: int, total_generations: int):
        """Enhanced dynamic weight adjustment for late-stage performance"""
        progress = min(generation / total_generations, 1.0)
        
        # Phase-based weight adjustment
        if progress < 0.3:  # Early phase - focus on accuracy
            accuracy_weight = 0.8
            robustness_weight = 0.05
            generalization_weight = 0.05
            speed_weight = 0.05
            size_weight = 0.05
        elif progress < 0.7:  # Mid phase - balanced
            accuracy_weight = 0.5
            robustness_weight = 0.15
            generalization_weight = 0.15
            speed_weight = 0.1
            size_weight = 0.1
        else:  # Late phase - focus on robustness and generalization
            accuracy_weight = 0.3
            robustness_weight = 0.25
            generalization_weight = 0.25
            speed_weight = 0.1
            size_weight = 0.1
        
        # Apply weights
        if FitnessObjective.ACCURACY in self.weights:
            self.weights[FitnessObjective.ACCURACY] = accuracy_weight
            
        if FitnessObjective.SPEED in self.weights:
            self.weights[FitnessObjective.SPEED] = speed_weight
            
        if FitnessObjective.SIZE in self.weights:
            self.weights[FitnessObjective.SIZE] = size_weight
            
        if FitnessObjective.ROBUSTNESS in self.weights:
            self.weights[FitnessObjective.ROBUSTNESS] = robustness_weight
            
        if FitnessObjective.GENERALIZATION in self.weights:
            self.weights[FitnessObjective.GENERALIZATION] = generalization_weight
    
    def get_pareto_front(self, population: List[np.ndarray], x_data, y_data) -> List[int]:
        """Find Pareto-optimal solutions in the population"""
        scores = []
        
        for individual in population:
            individual_scores = []
            for obj in self.objectives:
                if obj == FitnessObjective.ACCURACY:
                    logits = mlp(individual, x_data)
                    individual_scores.append(float(get_acc(logits, y_data)))
                # Add other objectives as needed
            scores.append(individual_scores)
        
        scores = np.array(scores)
        pareto_front = []
        
        for i in range(len(scores)):
            is_pareto = True
            for j in range(len(scores)):
                if i != j and all(scores[j] >= scores[i]) and any(scores[j] > scores[i]):
                    is_pareto = False
                    break
            if is_pareto:
                pareto_front.append(i)
        
        return pareto_front

    def _load_diffusion_model(self, model_id: str):
        """Load a diffusion model and extract its UNet or text encoder"""
        if not DIFFUSERS_AVAILABLE:

            return None
        
        if self.use_cache and self.loaded_models and model_id in self.loaded_models:
            print(f"[OK] Diffusion model {model_id} already loaded from cache")
            return self.loaded_models[model_id]
        
        try:
            from diffusers import StableDiffusionPipeline, DiffusionPipeline
            
            print(f" Loading diffusion model {model_id}...")
            
            try:
                pipe = StableDiffusionPipeline.from_pretrained(
                    model_id,
                    cache_dir=self.cache_dir,
                    torch_dtype=torch.float16 if self.device == 'cuda' else torch.float32,
                    safety_checker=None,

                    model_id,
                    cache_dir=self.cache_dir,
                    torch_dtype=torch.float16 if self.device == 'cuda' else torch.float32
                )

            if hasattr(pipe, 'unet'):
                model = pipe.unet

                model.eval()
                if self.use_cache and self.loaded_models is not None:
                    self.loaded_models[model_id] = model
                num_params = sum(p.numel() for p in model.parameters())
                print(f"[OK] Loaded diffusion UNet from {model_id}: {num_params:,} parameters")
                return model

                model = pipe.text_encoder

                num_params = sum(p.numel() for p in model.parameters())

    def extract_parameters(self, model) -> np.ndarray:
        """Extract model parameters as a flat numpy array"""
        if model is None:
            return None
        params = []
        for p in model.parameters():
            params.append(p.detach().cpu().numpy().flatten())
        return np.concatenate(params)
    
    def create_mlp_from_hf(self, model, use_architecture_matching: bool = True) -> np.ndarray:
        """Convert HF model to our MLP format with intelligent mapping"""
        if model is None:
            if JAX_AVAILABLE:
                return jax.random.normal(jax.random.PRNGKey(0), (num_params,)) * 0.01
            else:
                return np.random.randn(num_params) * 0.01
        
        params = self.extract_parameters(model)
        
        if use_architecture_matching:
            dummy_mlp = self._create_dummy_mlp()
            params, _ = self.architecture_mapper.align_parameters(
                params, np.zeros(num_params), model, dummy_mlp
            )
        
        # Ensure correct size
        if len(params) < num_params:
            padding = np.random.randn(num_params - len(params)) * 0.001
            params = np.concatenate([params, padding])
        elif len(params) > num_params:
            importance = np.abs(params)
            if len(importance) > num_params:
                threshold = np.sort(importance)[-num_params]
                mask = importance >= threshold
                important_params = params[mask][:num_params]
                params = important_params
            else:
                params = params[:num_params]
        
        # GPU 메모리 해제 - use_cache=False 일 때 즉시 해제
        if not self.use_cache:
            del model
            gc.collect()
            if torch.cuda.is_available():
                torch.cuda.empty_cache()
        
        return params
    
    def _create_dummy_mlp(self):
        """Create a dummy MLP model for architecture matching"""
        class DummyMLP(nn.Module):
            def __init__(self):
                super().__init__()
                self.fc1 = nn.Linear(input_dim, hidden_dim)
                self.fc2 = nn.Linear(hidden_dim, output_dim)
        
        return DummyMLP()
    
    def merge_models_advanced(self, params1: np.ndarray, params2: np.ndarray, 
                            model1=None, model2=None, method: str = "slerp", 
                            ratio: float = 0.5) -> np.ndarray:
        """Advanced model merging with multiple strategies"""
        
        # Align parameters if models provided
        if model1 is not None and model2 is not None:
            params1, params2 = self.architecture_mapper.align_parameters(
                params1, params2, model1, model2
            )
        
        if method == "slerp":
            return slerp(ratio, params1, params2)
        elif method == "linear":
            return ratio * params2 + (1 - ratio) * params1
        elif method == "fisher_weighted":
            fisher1 = np.abs(params1) + 0.1
            fisher2 = np.abs(params2) + 0.1
            total_fisher = fisher1 + fisher2
            weights1 = fisher1 / total_fisher
            weights2 = fisher2 / total_fisher
            return weights1 * params1 + weights2 * params2
        else:
            return slerp(ratio, params1, params2)

def load_data(dataset_type="digits", train_test_split=0.8):
    """Load dataset - digits or MNIST"""
    if dataset_type == "digits":
        from sklearn import datasets
        images, labels = datasets.load_digits(return_X_y=True)
        x_train = images[:int(len(images) * train_test_split)]
        y_train = labels[:int(len(images) * train_test_split)]
        x_test = images[int(len(images) * train_test_split):]
        y_test = labels[int(len(images) * train_test_split):]
    else:  # MNIST
        try:
            from torchvision import datasets, transforms
            transform = transforms.Compose([
                transforms.ToTensor(),
                transforms.Normalize((0.1307,), (0.3081,))
            ])
            
            train_dataset = datasets.MNIST('data', train=True, download=True, transform=transform)
            test_dataset = datasets.MNIST('data', train=False, transform=transform)
            
            x_train = train_dataset.data.numpy().reshape(-1, 784)
            y_train = train_dataset.targets.numpy()
            x_test = test_dataset.data.numpy().reshape(-1, 784)
            y_test = test_dataset.targets.numpy()
            
            # Resize to 64 dims (8x8) for compatibility
            from sklearn.decomposition import PCA
            pca = PCA(n_components=64)
            x_train = pca.fit_transform(x_train)
            x_test = pca.transform(x_test)
            
            # Limit size
            x_train = x_train[:5000]
            y_train = y_train[:5000]
            x_test = x_test[:1000]
            y_test = y_test[:1000]
        except:
            print("[WARN] Could not load MNIST, using sklearn digits")
            return load_data("digits", train_test_split)
    
    return (x_train, y_train), (x_test, y_test)

@jax.jit if JAX_AVAILABLE else lambda f: f
def mlp(params, data):
    """A 2-layer feedforward network."""
    if JAX_AVAILABLE:
        ss = 0
        ee = hidden_dim * input_dim
        w = params[ss:ee].reshape([input_dim, hidden_dim])
        ss = ee
        ee = ss + hidden_dim
        b = params[ss:ee]
        x = jax.nn.relu(jnp.matmul(data, w) + b[None, ...])
        
        ss = ee
        ee = ss + output_dim * hidden_dim
        w = params[ss:ee].reshape([hidden_dim, output_dim])
        ss = ee
        ee = ss + output_dim
        b = params[ss:ee]
        x = jnp.matmul(x, w) + b[None, ...]
        return x
    else:
        # NumPy fallback
        ss = 0
        ee = hidden_dim * input_dim
        w = params[ss:ee].reshape([input_dim, hidden_dim])
        ss = ee
        ee = ss + hidden_dim
        b = params[ss:ee]
        x = np.maximum(0, np.matmul(data, w) + b[None, ...])  # ReLU
        
        ss = ee
        ee = ss + output_dim * hidden_dim
        w = params[ss:ee].reshape([hidden_dim, output_dim])
        ss = ee
        ee = ss + output_dim
        b = params[ss:ee]
        x = np.matmul(x, w) + b[None, ...]
        return x

@jax.jit if JAX_AVAILABLE else lambda f: f
def get_acc(logits, labels):
    """Calculate accuracy"""
    if JAX_AVAILABLE:
        predicted_labels = jnp.argmax(logits, axis=1)
        return jnp.mean(predicted_labels == labels)
    else:
        predicted_labels = np.argmax(logits, axis=1)
        return np.mean(predicted_labels == labels)

def get_loss(params, data, labels):
    """Calculate loss"""
    logits = mlp(params, data)
    if JAX_AVAILABLE:
        log_probs = log_softmax(logits)
        true_log_probs = jnp.take_along_axis(log_probs, labels[:, None], axis=1)
        return -jnp.mean(true_log_probs)
    else:
        # NumPy fallback
        exp_logits = np.exp(logits - np.max(logits, axis=1, keepdims=True))
        probs = exp_logits / np.sum(exp_logits, axis=1, keepdims=True)
        log_probs = np.log(probs + 1e-10)
        true_log_probs = log_probs[np.arange(len(labels)), labels]
        return -np.mean(true_log_probs)

if JAX_AVAILABLE:
    get_grad = jax.grad(get_loss)

def train(params, x, y, lr=0.003, num_epochs=10):
    """Train the model"""
    if JAX_AVAILABLE:
        opt = optax.adam(lr)
        opt_state = opt.init(params)
        
        @jax.jit
        def train_step(params, opt_state, x, y):
            grad = get_grad(params, x, y)
            updates, opt_state = opt.update(grad, opt_state)
            params = optax.apply_updates(params, updates)
            return params, opt_state
        
        for _ in range(num_epochs):
            params, opt_state = train_step(params, opt_state, x, y)
        return params
    else:
        # PyTorch autograd 기반 fallback (수치미분 대비 수천 배 빠름)
        params_tensor = torch.tensor(params, dtype=torch.float32, requires_grad=True)
        x_tensor = torch.tensor(x, dtype=torch.float32)
        y_tensor = torch.tensor(y, dtype=torch.long)
        optimizer = torch.optim.Adam([params_tensor], lr=lr)
        
        for _ in range(num_epochs):
            optimizer.zero_grad()
            # MLP forward (torch 버전)
            ss, ee = 0, hidden_dim * input_dim
            w1 = params_tensor[ss:ee].reshape(input_dim, hidden_dim)
            ss, ee = ee, ee + hidden_dim
            b1 = params_tensor[ss:ee]
            h = torch.relu(x_tensor @ w1 + b1.unsqueeze(0))
            
            ss, ee = ee, ee + output_dim * hidden_dim
            w2 = params_tensor[ss:ee].reshape(hidden_dim, output_dim)
            ss, ee = ee, ee + output_dim
            b2 = params_tensor[ss:ee]
            logits = h @ w2 + b2.unsqueeze(0)
            
            loss = torch.nn.functional.cross_entropy(logits, y_tensor)
            loss.backward()
            optimizer.step()
        
        return params_tensor.detach().numpy()

def slerp(val, x, y):
    """Spherical linear interpolation"""
    if JAX_AVAILABLE:
        # Normalize the inputs
        norm_x = x / jnp.linalg.norm(x)
        norm_y = y / jnp.linalg.norm(y)
        
        # Cosine of the angle
        dot = jnp.dot(norm_x, norm_y)
        omega = jnp.arccos(jnp.clip(dot, -1, 1))
        sin_omega = jnp.sin(omega)
        
        # Calculate scales for input vectors
        scale_x = jnp.sin((1.0 - val) * omega) / sin_omega
        scale_y = jnp.sin(val * omega) / sin_omega
        
        # Linear interpolation weights
        lin_scale_x = 1.0 - val
        lin_scale_y = val
        
        return jnp.where(
            sin_omega > 1e-6,
            scale_x * x + scale_y * y,
            lin_scale_x * x + lin_scale_y * y
        )
    else:
        # NumPy version
        norm_x = x / np.linalg.norm(x)
        norm_y = y / np.linalg.norm(y)
        dot = np.dot(norm_x, norm_y)
        omega = np.arccos(np.clip(dot, -1, 1))
        sin_omega = np.sin(omega)
        
        if sin_omega > 1e-6:
            scale_x = np.sin((1.0 - val) * omega) / sin_omega
            scale_y = np.sin(val * omega) / sin_omega
            return scale_x * x + scale_y * y
        else:
            return (1.0 - val) * x + val * y

def slerp_w_splitpoint(val, split_point, x, y):
    """SLERP with splitpoint"""
    if JAX_AVAILABLE:
        val = jnp.ones_like(x) * val
        mask = jnp.arange(len(val)) < split_point
        val = jnp.where(mask, val, 1 - val)
        
        norm_x = x / jnp.linalg.norm(x)
        norm_y = y / jnp.linalg.norm(y)
        dot = jnp.dot(norm_x, norm_y)
        omega = jnp.arccos(jnp.clip(dot, -1, 1))
        sin_omega = jnp.sin(omega)
        
        scale_x = jnp.sin((1.0 - val) * omega) / sin_omega
        scale_y = jnp.sin(val * omega) / sin_omega
        lin_scale_x = 1.0 - val
        lin_scale_y = val
        
        return jnp.where(
            sin_omega > 1e-6,
            scale_x * x + scale_y * y,
            lin_scale_x * x + lin_scale_y * y
        )
    else:
        val = np.ones_like(x) * val
        mask = np.arange(len(val)) < split_point
        val = np.where(mask, val, 1 - val)
        
        norm_x = x / np.linalg.norm(x)
        norm_y = y / np.linalg.norm(y)
        dot = np.dot(norm_x, norm_y)
        omega = np.arccos(np.clip(dot, -1, 1))
        sin_omega = np.sin(omega)
        
        if sin_omega > 1e-6:
            scale_x = np.sin((1.0 - val) * omega) / sin_omega
            scale_y = np.sin(val * omega) / sin_omega
            return scale_x * x + scale_y * y
        else:
            return (1.0 - val) * x + val * y

@jax.jit if JAX_AVAILABLE else lambda f: f
def crossover(parents, rand_key=None):
    """Crossover with splitpoint"""
    if JAX_AVAILABLE and rand_key is not None:
        k1, k2 = jax.random.split(rand_key)
        split_point = jax.random.randint(k1, shape=(), minval=0, maxval=parents[0].shape[0])
        w = jax.random.uniform(k2)
        return slerp_w_splitpoint(w, split_point, parents[0], parents[1])
    else:
        split_point = np.random.randint(0, parents[0].shape[0])
        w = np.random.uniform()
        return slerp_w_splitpoint(w, split_point, parents[0], parents[1])

@jax.jit if JAX_AVAILABLE else lambda f: f
def crossover_without_splitpoint(parents, rand_key=None):
    """Uniform crossover without splitpoint"""
    if JAX_AVAILABLE and rand_key is not None:
        w = jax.random.uniform(rand_key)
        return slerp(w, parents[0], parents[1])
    else:
        w = np.random.uniform()
        return slerp(w, parents[0], parents[1])

def sample_parents_with_novelty(archive, scores, rand_key, alpha, use_matchmaker, 
                                novelty_lambda=0.0):
    """Enhanced parent selection with novelty-based diversity maintenance"""
    if JAX_AVAILABLE and rand_key is not None:
        k1, k2 = jax.random.split(rand_key)
        z = scores.sum(axis=0)
        z = jnp.where(z, z, 1) ** alpha
        fitness_matrix = scores / z[None, :]
        fitness = jnp.sum(fitness_matrix, axis=1)
        
        # Compute novelty scores if lambda > 0
        if novelty_lambda > 0:
            # Compute pairwise diversity
            diversity_scores = []
            for i in range(len(archive)):
                # Sample random individuals for diversity comparison
                sample_idx = jax.random.choice(k1, len(archive), shape=(min(10, len(archive)),))
                diversities = [compute_cosine_diversity(archive[i], archive[j]) 
                              for j in sample_idx if j != i]
                diversity_scores.append(np.mean(diversities) if diversities else 0.0)
            diversity_scores = jnp.array(diversity_scores)
            
            # Normalize diversity scores
            if jnp.max(diversity_scores) > 0:
                diversity_scores = diversity_scores / jnp.max(diversity_scores)
            
            # Combine fitness and diversity
            probs = fitness * (1.0 + novelty_lambda * diversity_scores)
        else:
            probs = fitness
        
        probs = probs / jnp.sum(probs)
        
        if use_matchmaker:
            parent_1_idx = jax.random.choice(k1, probs.size, shape=(1,), p=probs)[0]
            match_score = jnp.maximum(0, fitness_matrix - fitness_matrix[parent_1_idx, :]).sum(axis=1)
            probs = match_score / jnp.sum(match_score)
            parent_2_idx = jax.random.choice(k2, probs.size, shape=(1,), p=probs)[0]
        else:
            parent_2_idx, parent_1_idx = jax.random.choice(k1, probs.size, shape=(2,), p=probs)
    else:
        z = scores.sum(axis=0)
        z = np.where(z, z, 1) ** alpha
        fitness_matrix = scores / z[None, :]
        fitness = np.sum(fitness_matrix, axis=1)
        
        # Compute novelty scores if lambda > 0
        if novelty_lambda > 0:
            diversity_scores = []
            for i in range(len(archive)):
                sample_idx = np.random.choice(len(archive), 
                                            min(10, len(archive)), replace=False)
                diversities = [compute_cosine_diversity(archive[i], archive[j]) 
                              for j in sample_idx if j != i]
                diversity_scores.append(np.mean(diversities) if diversities else 0.0)
            diversity_scores = np.array(diversity_scores)
            
            if np.max(diversity_scores) > 0:
                diversity_scores = diversity_scores / np.max(diversity_scores)
            
            probs = fitness * (1.0 + novelty_lambda * diversity_scores)
        else:
            probs = fitness
        
        probs = probs / np.sum(probs)
        
        if use_matchmaker:
            parent_1_idx = np.random.choice(probs.size, p=probs)
            match_score = np.maximum(0, fitness_matrix - fitness_matrix[parent_1_idx, :]).sum(axis=1)
            probs = match_score / np.sum(match_score)
            parent_2_idx = np.random.choice(probs.size, p=probs)
        else:
            parent_1_idx, parent_2_idx = np.random.choice(probs.size, size=2, p=probs)
    
    return archive[parent_1_idx], archive[parent_2_idx], parent_1_idx, parent_2_idx

def update_archive(score, param, archive, scores, alpha):
    """Update archive with new individual"""
    if JAX_AVAILABLE:
        ext_scores = jnp.concatenate([scores, score[None, ...]], axis=0)
        z = jnp.sum(ext_scores, axis=0) ** alpha
        z = jnp.where(z, z, 1)
        ext_scores /= z[None, :]
        fitness = jnp.sum(ext_scores, axis=1)
        worst_ix = jnp.argmin(fitness)
        update_mask = worst_ix < scores.shape[0]
        
        scores = scores.at[worst_ix].set(
            jax.lax.select(update_mask, score, scores[worst_ix])
        )
        archive = archive.at[worst_ix].set(
            jax.lax.select(update_mask, param, archive[worst_ix])
        )
    else:
        ext_scores = np.concatenate([scores, score[None, ...]], axis=0)
        z = np.sum(ext_scores, axis=0) ** alpha
        z = np.where(z, z, 1)
        ext_scores /= z[None, :]
        fitness = np.sum(ext_scores, axis=1)
        worst_ix = np.argmin(fitness)
        
        if worst_ix < scores.shape[0]:
            scores[worst_ix] = score
            archive[worst_ix] = param
    
    return archive, scores

# Enhanced Natural Niches Algorithm WITH LINEAGE TRACKING and Late-Stage Improvements
# ============================================
def run_natural_niches_enhanced(
    runs: int,
    pop_size: int,
    total_forward_passes: int,
    store_train_results: bool,
    no_matchmaker: bool,
    no_crossover: bool,
    no_splitpoint: bool,
    use_pre_trained: bool,
    alpha: float = 1.0,
    model1_hf=None,
    model2_hf=None,
    model1_file=None,
    model2_file=None,
    merge_ratio=0.5,
    merge_method="slerp",
    fitness_objectives=None,
    fitness_weights=None,
    track_lineage: bool = True
) -> List[Dict]:
    """Run Enhanced Natural Niches algorithm with adaptive fitness and lineage tracking"""
    
    (x_train, y_train), (x_test, y_test) = load_data()
    use_matchmaker = not no_matchmaker
    use_crossover = not no_crossover
    use_splitpoint = not no_splitpoint
    
    # Initialize adaptive fitness evaluator
    if fitness_objectives is None:
        fitness_objectives = [FitnessObjective.ACCURACY]
    if fitness_weights is None:
        fitness_weights = {FitnessObjective.ACCURACY: 1.0}
    
    fitness_evaluator = AdaptiveFitnessEvaluator(fitness_objectives, fitness_weights)
    
    # Initialize lineage tracker
    lineage_tracker = LineageTracker() if track_lineage else None
    
    results = []
    
    if use_pre_trained:
        model_1, model_2 = get_pre_trained_models(
            model1_hf, model2_hf, model1_file, model2_file, merge_ratio, merge_method
        )
    
    for run in tqdm(range(runs), desc="Runs"):
        result = defaultdict(list)
        seed = 42 + run
        
        # Initialize adaptive parameters
        std_base = 0.01
        std_current = std_base
        improvement_window = deque(maxlen=50)  # Track last 50 improvements
        
        if JAX_AVAILABLE:
            key = jax.random.PRNGKey(seed)
            archive = jnp.zeros([pop_size, num_params])
            scores = jnp.zeros([pop_size, len(x_train)], dtype=jnp.bool_)
        else:
            np.random.seed(seed)
            archive = np.zeros([pop_size, num_params])
            scores = np.zeros([pop_size, len(x_train)], dtype=bool)
        
        if not use_pre_trained:
            if JAX_AVAILABLE:
                key, key1, key2 = jax.random.split(key, 3)
                model_1 = jax.random.normal(key1, (num_params,)) * 0.01
                model_2 = jax.random.normal(key2, (num_params,)) * 0.01
            else:
                model_1 = np.random.randn(num_params) * 0.01
                model_2 = np.random.randn(num_params) * 0.01
        
        # Initialize archive with two models
        for model in (model_1, model_2):
            logits = mlp(model, x_train)
            if JAX_AVAILABLE:
                score = jnp.argmax(logits, axis=1) == y_train
            else:
                score = np.argmax(logits, axis=1) == y_train
            archive, scores = update_archive(score, model, archive, scores, alpha)
            
            # Track initial models
            if lineage_tracker:
                model_id = lineage_tracker.get_model_id(model)
                metrics = comprehensive_evaluation(model, x_train, y_train, x_test, y_test)
                lineage_tracker.record_performance(model_id, metrics)
        
        # Evolution loop with late-stage enhancements
        for i in tqdm(range(total_forward_passes), desc="Forward passes", leave=False):
            progress = i / total_forward_passes
            
            # 1. Update fitness weights dynamically (Enhanced Phase-Based)
            fitness_evaluator.update_weights_dynamically(i, total_forward_passes)
            
            # 2. Update adaptive mutation std
            if i > 50 and i % 25 == 0:
                if len(improvement_window) >= 25:
                    recent_improvements = list(improvement_window)[-25:]
                    avg_improvement = np.mean(recent_improvements)
                    
                    if avg_improvement < 0.001:  # Stagnation
                        std_current = min(std_current * 1.2, 0.1)
                        if i % 100 == 0:
                            print(f"  Stagnation detected at {i}, increasing std to {std_current:.4f}")
                    else:  # Good progress
                        std_current = max(std_current * 0.95, 0.002)
            
            # 3. Determine novelty weight based on progress
            if progress < 0.5:
                novelty_lambda = 0.0  # Early phase - pure fitness
            elif progress < 0.7:
                novelty_lambda = 0.2  # Mid phase - some diversity
            else:
                novelty_lambda = 0.4  # Late phase - strong diversity pressure
            
            if JAX_AVAILABLE:
                k1, k2, k3, key = jax.random.split(key, 4)
            else:
                k1 = k2 = k3 = None
            
            # Sample parents with novelty weighting
            parent1, parent2, p1_idx, p2_idx = sample_parents_with_novelty(
                archive, scores, k1, alpha, use_matchmaker, novelty_lambda
            )
            
            # Check parent diversity and resample if too similar
            if progress > 0.5:
                parent_diversity = compute_cosine_diversity(parent1, parent2)
                if parent_diversity < 0.1:  # Too similar
                    # Force diversity by resampling
                    if JAX_AVAILABLE:
                        k_resample, key = jax.random.split(key)
                        p2_idx = jax.random.randint(k_resample, shape=(), 
                                                   minval=0, maxval=pop_size)
                    else:
                        p2_idx = np.random.randint(0, pop_size)
                    parent2 = archive[p2_idx]
            
            # Create child with advanced merging in late stage
            if use_crossover:
                # Use fisher-weighted merging in late stage
                if progress > 0.7 and np.random.random() < 0.3:
                    # 30% chance of fisher-weighted merge in late stage
                    child = enhanced_model_manager.merge_models_advanced(
                        parent1, parent2, method="fisher_weighted", 
                        ratio=np.random.uniform(0.3, 0.7)
                    )
                else:
                    if use_splitpoint:
                        child = crossover([parent1, parent2], k2)
                    else:
                        child = crossover_without_splitpoint([parent1, parent2], k2)
            else:
                child = parent1.copy()
            
            # Apply adaptive mutation
            if not use_pre_trained:
                child = mutate(child, k3, std=std_current)
            
            # 4. Apply Lamarckian refinement to promising individuals in late stage
            if progress > 0.7 and len(result["test_values"]) > 100:
                # Check if child is promising (top 20% of recent performance)
                recent_fitness = result["multi_objective_fitness"][-100:] if "multi_objective_fitness" in result else []
                if recent_fitness:
                    threshold = np.percentile(recent_fitness, 80)
                    child_fitness = fitness_evaluator.evaluate(child, x_train, y_train, x_test, y_test)
                    
                    if child_fitness > threshold:
                        # Apply lightweight fine-tuning
                        child = apply_lamarckian_refinement(
                            child, x_train, y_train, 
                            subset_size=256, lr=0.001, epochs=2
                        )
                        if i % 500 == 0:
                            print(f"  Applied Lamarckian refinement at step {i}")
            
            # Track lineage
            if lineage_tracker:
                child_id = lineage_tracker.record_birth(child, parent1, parent2, i)
                
                # Evaluate child comprehensively
                child_metrics = comprehensive_evaluation(child, x_train, y_train, x_test, y_test)
                lineage_tracker.record_performance(child_id, child_metrics)
                
                # Calculate improvement
                improvements = lineage_tracker.evaluate_improvement(child_id)
                if improvements and improvements.get('test_acc', 0) > 5:
                    print(f" Generation {i}: 부모 대비 {improvements['test_acc']:.2f}% 개선!")
            
            # Use adaptive fitness evaluation
            fitness = fitness_evaluator.evaluate(child, x_train, y_train, x_test, y_test)
            
            logits = mlp(child, x_train)
            if JAX_AVAILABLE:
                score = jnp.argmax(logits, axis=1) == y_train
            else:
                score = np.argmax(logits, axis=1) == y_train
            archive, scores = update_archive(score, child, archive, scores, alpha)
            
            # Log results
            result["evals"].append(i)
            result["multi_objective_fitness"].append(fitness)
            train_acc = scores.mean(axis=1)
            
            if store_train_results:
                result["train_values"].append(float(train_acc.max()))
            
            # Test accuracy
            best_individual = train_acc.argmax()
            logits = mlp(archive[best_individual], x_test)
            acc = get_acc(logits, y_test)
            result["test_values"].append(float(acc))
            
            # Track improvement for adaptive std
            if len(result["test_values"]) > 1:
                improvement = result["test_values"][-1] - result["test_values"][-2]
                improvement_window.append(improvement)
            
            # Store additional metrics
            for obj in fitness_objectives:
                if obj in fitness_evaluator.history:
                    result[f"fitness_{obj.value}"].append(fitness_evaluator.history[obj][-1])
            
            if i % 1000 == 0:
                print(f"Run {run}, Step {i}, Test acc: {acc:.4f}, Multi-obj fitness: {fitness:.4f}, Std: {std_current:.4f}")
        
        # Find Pareto front at the end
        pareto_indices = fitness_evaluator.get_pareto_front(archive, x_train, y_train)
        result["pareto_front_size"] = len(pareto_indices)
        
        # Add lineage validation results
        if lineage_tracker:
            result["lineage_validation"] = lineage_tracker.get_statistical_validation()
            result["total_improvements"] = len([imp for imp in lineage_tracker.improvement_history 
                                               if imp['improvements'].get('test_acc', 0) > 0])
        
        results.append(dict(result))
    
    return results

# MAP-Elites Algorithm (Enhanced with Lineage Tracking and Late-Stage Improvements)

def update_archive_map_elites(
    behavior_descriptors: tuple,
    quality: float,
    param,
    archive,
    qualities,
    occupied,
):
    """Update MAP-Elites archive"""
    odd_acc, even_acc = behavior_descriptors
    
    if JAX_AVAILABLE:
        ix0 = jnp.floor(odd_acc * 10).astype(jnp.int32)
        ix1 = jnp.floor(even_acc * 10).astype(jnp.int32)
        ix0 = jnp.clip(ix0, 0, 9)
        ix1 = jnp.clip(ix1, 0, 9)
        
        update_mask = jnp.logical_or(~occupied[ix0, ix1], quality > qualities[ix0, ix1])
        occupied = occupied.at[ix0, ix1].set(
            jax.lax.select(update_mask, True, occupied[ix0, ix1])
        )
        qualities = qualities.at[ix0, ix1].set(
            jax.lax.select(update_mask, quality, qualities[ix0, ix1])
        )
        archive = archive.at[ix0, ix1].set(
            jax.lax.select(update_mask, param, archive[ix0, ix1])
        )
    else:
        ix0 = int(np.floor(odd_acc * 10))
        ix1 = int(np.floor(even_acc * 10))
        ix0 = min(max(ix0, 0), 9)
        ix1 = min(max(ix1, 0), 9)
        
        if not occupied[ix0, ix1] or quality > qualities[ix0, ix1]:
            occupied[ix0, ix1] = True
            qualities[ix0, ix1] = quality
            archive[ix0, ix1] = param
    
    return archive, qualities, occupied

def sample_parents_map_elites(archive, occupied, qualities, rand_key):
    """Sample parents from MAP-Elites archive"""
    if JAX_AVAILABLE and rand_key is not None:
        indices = jnp.argwhere(occupied)
        if indices.shape[0] == 0:
            return None
        flat_qualities = qualities[occupied]
        probs = flat_qualities / jnp.sum(flat_qualities)
        sampled_indices = jax.random.choice(
            rand_key,
            flat_qualities.size,
            shape=(2,),
            p=probs,
            replace=indices.shape[0] < 2,
        )
        sampled_positions = indices[sampled_indices]
        return archive[sampled_positions[:, 0], sampled_positions[:, 1]]
    else:
        indices = np.argwhere(occupied)
        if len(indices) == 0:
            return None
        flat_qualities = qualities[occupied]
        probs = flat_qualities / np.sum(flat_qualities)
        sampled_indices = np.random.choice(
            len(indices),
            size=2,
            p=probs,
            replace=len(indices) < 2,
        )
        sampled_positions = indices[sampled_indices]
        return archive[sampled_positions[:, 0], sampled_positions[:, 1]]

def run_cma_es(
    runs: int,
    pop_size: int,
    total_forward_passes: int,
    store_train_results: bool
) -> List[Dict]:
    """Run CMA-ES algorithm"""
    
    if not CMA_AVAILABLE:
        print("[WARN] CMA-ES not available, returning dummy results")
        return [{"error": "CMA library not installed"}]
    
    (x_train, y_train), (x_test, y_test) = load_data()
    results = []
    
    # Adjust total_forward_passes to be divisible by pop_size
    if total_forward_passes % pop_size != 0:
        old_total = total_forward_passes
        total_forward_passes = (total_forward_passes // pop_size) * pop_size
        print(f"[WARN] Adjusted total_forward_passes from {old_total} to {total_forward_passes} to be divisible by pop_size ({pop_size})")
    
    for run in tqdm(range(runs), desc="Runs"):
        result = defaultdict(list)
        seed = 42 + run
        
        if JAX_AVAILABLE:
            key = jax.random.PRNGKey(seed)
            rand_params = jax.random.normal(key, shape=(num_params,)) * 0.01
            
            v_mlp = jax.vmap(mlp, in_axes=(0, None))
            v_get_acc = jax.vmap(get_acc, in_axes=(0, None))
        else:
            np.random.seed(seed)
            rand_params = np.random.randn(num_params) * 0.01
        
        solver = cma.CMAEvolutionStrategy(
            x0=rand_params,
            sigma0=0.01,
            inopts={
                "popsize": pop_size,
                "seed": seed,
                "randn": np.random.randn,
            },
        )
        
        for i in tqdm(range(total_forward_passes // pop_size), desc="Generations", leave=False):
            solutions = solver.ask()
            
            if JAX_AVAILABLE:
                batch_logits = v_mlp(jnp.array(solutions), x_train)
                acc_batch = v_get_acc(batch_logits, y_train)
            else:
                acc_batch = []
                for sol in solutions:
                    logits = mlp(sol, x_train)
                    acc = get_acc(logits, y_train)
                    acc_batch.append(acc)
                acc_batch = np.array(acc_batch)
            
            neg_fitness = -acc_batch
            solver.tell(solutions, neg_fitness.tolist())
            
            result["evals"].append(i * pop_size)
            
            if store_train_results:
                result["train_values"].append(float(-solver.result.fbest))
            
            # Test
            if JAX_AVAILABLE:
                logits = mlp(jnp.array(solver.result.xbest), x_test)
            else:
                logits = mlp(np.array(solver.result.xbest), x_test)
            
            acc = get_acc(logits, y_test)
            result["test_values"].append(float(acc))
            
            if i % 10 == 0:
                print(f"Run {run}, Gen {i}, Best Acc: {-solver.result.fbest:.4f}")
        
        results.append(dict(result))
    
    return results

# Brute Force Algorithm (unchanged)
# ============================================
def run_brute_force_enhanced(
    model1_hf=None,
    model2_hf=None,
    model1_file=None,
    model2_file=None,
    merge_method="slerp",
    fitness_objectives=None,
    fitness_weights=None
):
    """Run enhanced brute force search with adaptive fitness"""
    
    (x_train, y_train), (x_test, y_test) = load_data()
    
    # Initialize adaptive fitness evaluator
    if fitness_objectives is None:
        fitness_objectives = [FitnessObjective.ACCURACY]
    if fitness_weights is None:
        fitness_weights = {FitnessObjective.ACCURACY: 1.0}
    
    fitness_evaluator = AdaptiveFitnessEvaluator(fitness_objectives, fitness_weights)
    
    model_1, model_2 = get_pre_trained_models(
        model1_hf, model2_hf, model1_file, model2_file, merge_method=merge_method
    )
    
    best_fitness = -np.inf if not JAX_AVAILABLE else -jnp.inf
    best_w = None
    best_model = None
    
    # Search over interpolation weights
    if JAX_AVAILABLE:
        weights = jnp.linspace(0, 1, 10000)
    else:
        weights = np.linspace(0, 1, 10000)
    
    fitness_history = []
    
    print(" Searching for best interpolation weight with adaptive fitness...")
    for w in tqdm(weights, desc="Testing weights"):
        if merge_method == "slerp":
            model = slerp(w, model_1, model_2)
        elif merge_method == "linear":
            model = w * model_2 + (1 - w) * model_1
        else:
            model = slerp(w, model_1, model_2)
        
        # Evaluate with adaptive fitness
        fitness = fitness_evaluator.evaluate(model, x_train, y_train, x_test, y_test)
        fitness_history.append(fitness)
        
        if fitness > best_fitness:
            best_fitness = fitness
            best_w = w
            best_model = model
    
    # Test best model
    logits = mlp(best_model, x_test)
    test_acc = get_acc(logits, y_test)
    
    # Get detailed metrics for best model
    detailed_metrics = {}
    for obj in fitness_objectives:
        if obj in fitness_evaluator.history:
            detailed_metrics[obj.value] = fitness_evaluator.history[obj][-1]
    
    result = {
        "best_w": float(best_w),
        "test_acc": float(test_acc),
        "best_fitness": float(best_fitness),
        "evaluated_models": 10000,
        "best_model_performance": float(test_acc),
        "detailed_metrics": detailed_metrics,
        "evals": list(range(0, 10000, 100)),
        "test_values": [float(test_acc)] * 100,
        "fitness_history": fitness_history[::100]
    }
    
    print(f"[OK] Best weight: {best_w:.4f}, Test accuracy: {test_acc:.4f}, Fitness: {best_fitness:.4f}")
    
    return [result]

def create_enhanced_summary_table(results, method, fitness_objectives=None):
    """Create enhanced summary table with improved metrics"""
    try:
        summary_data = {"Metric": [], "Value": []}
        
        summary_data["Metric"].append("Method")
        summary_data["Value"].append(method.upper())
        
        if isinstance(results, list) and len(results) > 0:
            summary_data["Metric"].append("Number of Runs")
            summary_data["Value"].append(str(len(results)))
           
            # 초기 vs 최종 성능
            initial_performances = []
            final_performances = []
           
            for run_result in results:
                if 'test_values' in run_result and len(run_result['test_values']) > 0:
                    initial_vals = run_result['test_values'][:min(100, len(run_result['test_values']))]
                    initial_performances.append(np.mean(initial_vals))
                    
                    final_vals = run_result['test_values'][-min(100, len(run_result['test_values'])):]
                    final_performances.append(np.mean(final_vals))
           
            # 초기/최종 성능 메트릭
            if initial_performances and final_performances:
                init_mean = np.mean(initial_performances)
                final_mean = np.mean(final_performances)
                
                summary_data["Metric"].append("--- 전체 성능 ---")
                summary_data["Value"].append("")
                
                summary_data["Metric"].append("Initial Performance")
                summary_data["Value"].append(f"{init_mean:.4f} ({init_mean*100:.2f}%)")
                
                summary_data["Metric"].append("Final Performance")
                summary_data["Value"].append(f"{final_mean:.4f} ({final_mean*100:.2f}%)")
                
                summary_data["Metric"].append("Total Improvement")
                if init_mean > 0:
                    improvement = ((final_mean - init_mean) / init_mean) * 100
                    summary_data["Value"].append(f"+{improvement:.1f}%")
                else:
                    summary_data["Value"].append("N/A")
                
                summary_data["Metric"].append("Best Final Accuracy")
                summary_data["Value"].append(f"{np.max(final_performances):.4f}")
                
                summary_data["Metric"].append("Avg Final Accuracy")
                summary_data["Value"].append(f"{final_mean:.4f} ± {np.std(final_performances):.4f}")
           
            # Lineage 통계
            if 'lineage_validation' in results[0]:
                summary_data["Metric"].append("--- 수렴 분석 ---")
                summary_data["Value"].append("")
                
                val = results[0]['lineage_validation']
                if 'error' not in val:
                    summary_data["Metric"].append("후기 부모 성능")
                    summary_data["Value"].append(f"{val.get('mean_parent', 0):.4f}")
                    
                    summary_data["Metric"].append("후기 자식 성능")
                    summary_data["Value"].append(f"{val.get('mean_child', 0):.4f}")
                    
                    summary_data["Metric"].append("후기 수렴율")
                    convergence = val.get('improvement_percent', 0)
                    summary_data["Value"].append(f"{abs(convergence):.2f}%")
                    
                    summary_data["Metric"].append("통계적 유의성")
                    if val.get('significant', False):
                        summary_data["Value"].append(f"Yes (p={val.get('p_value', 1):.4f})")
                    else:
                        summary_data["Value"].append(f"No (p={val.get('p_value', 1):.4f})")
           
            # 개선 통계
            if 'total_improvements' in results[0]:
                summary_data["Metric"].append("--- 개선 통계 ---")
                summary_data["Value"].append("")
                
                total_imps = [r.get('total_improvements', 0) for r in results]
                summary_data["Metric"].append("Total Improvements")
                summary_data["Value"].append(f"{np.sum(total_imps):,} times")
                
                summary_data["Metric"].append("Avg Improvements/Run")
                summary_data["Value"].append(f"{np.mean(total_imps):.1f}")
           
            # Multi-objective 메트릭
            if 'multi_objective_fitness' in results[0]:
                final_fitness = []
                for r in results:
                    if r['multi_objective_fitness']:
                        final_fitness.append(r['multi_objective_fitness'][-1])
                
                if final_fitness:
                    summary_data["Metric"].append("--- Fitness ---")
                    summary_data["Value"].append("")
                    
                    summary_data["Metric"].append("Final Multi-Obj Fitness")
                    summary_data["Value"].append(f"{np.mean(final_fitness):.4f}")
           
            # MAP-Elites Coverage
            if method == "map_elites" and 'coverage' in results[0]:
                final_coverages = []
                for r in results:
                    if 'coverage' in r and r['coverage']:
                        final_coverages.append(r['coverage'][-1])
                
                if final_coverages:
                    summary_data["Metric"].append("Final Coverage")
                    summary_data["Value"].append(f"{np.mean(final_coverages):.1f}")
           
            # 진화 단계 설명
            summary_data["Metric"].append("--- 진화 단계 ---")
            summary_data["Value"].append("")
            
            summary_data["Metric"].append("초기 (0-20%)")
            summary_data["Value"].append("급격한 개선")
            
            summary_data["Metric"].append("중기 (20-60%)")
            summary_data["Value"].append("안정적 개선")
            
            summary_data["Metric"].append("후기 (60-100%)")
            summary_data["Value"].append("미세 조정/수렴")
       
        df = pd.DataFrame(summary_data)
        return df
       
    except Exception as e:
        df = pd.DataFrame({
            "Metric": ["Error"],
            "Value": [str(e)]
        })
        return df

def create_and_upload_merged_model(
    pkl_file_path: str,
    model1_hf: str,
    model2_hf: str,
    repo_name: str,
    private: bool = False,
    progress=gr.Progress()
) -> str:
    """Create and upload merged model to Hugging Face Hub"""
    try:
        # Check requirements
        if not HF_HUB_AVAILABLE:
            return "[FAIL] Hugging Face Hub API가 설치되지 않았습니다. pip install huggingface_hub"
        
        if not TRANSFORMERS_AVAILABLE:
            return "[FAIL] Transformers 라이브러리가 설치되지 않았습니다."
        
        # Get HF token
        hf_token = (os.getenv("HF_TOKEN") or "").strip()
        if not hf_token:
            return "[FAIL] HF_TOKEN 환경변수가 설정되지 않았습니다. export HF_TOKEN=your_token"
        
        progress(0.1, desc="실험 결과 로드 중...")
        
        # Load experiment results
        with open(pkl_file_path, 'rb') as f:
            results = pickle.load(f)
        
        # Extract best parameters
        if isinstance(results, list) and len(results) > 0:
            best_acc = 0
            best_run_idx = 0
            for idx, run in enumerate(results):
                if 'test_values' in run and len(run['test_values']) > 0:
                    final_acc = run['test_values'][-1]
                    if final_acc > best_acc:
                        best_acc = final_acc
                        best_run_idx = idx
            
            progress(0.3, desc=f"최적 모델 확인 (정확도: {best_acc:.2%})...")
        else:
            return "[FAIL] 실험 결과 파일 형식이 올바르지 않습니다."
        
        progress(0.4, desc="원본 모델 로드 중...")
        
        # Load and merge models (simplified)
        model1 = AutoModelForCausalLM.from_pretrained(
            model1_hf,
            torch_dtype=torch.float16 if torch.cuda.is_available() else torch.float32,
            token=hf_token
        )
        
        model2 = AutoModelForCausalLM.from_pretrained(
            model2_hf,
            torch_dtype=torch.float16 if torch.cuda.is_available() else torch.float32,
            token=hf_token
        )
        
        progress(0.6, desc="모델 병합 중...")
        
        optimal_ratio = 0.5
        with torch.no_grad():
            for p1, p2 in zip(model1.parameters(), model2.parameters()):
                if p1.shape == p2.shape:
                    p1.data = (1 - optimal_ratio) * p1.data + optimal_ratio * p2.data
        
        tokenizer = AutoTokenizer.from_pretrained(model1_hf, token=hf_token)
        
        progress(0.7, desc="임시 디렉토리 생성 중...")
        
        # Create temporary directory for model
        import tempfile
        
        with tempfile.TemporaryDirectory() as tmp_dir:
            # Save merged model
            model1.save_pretrained(tmp_dir)
            tokenizer.save_pretrained(tmp_dir)
            
            # Create model card
            model_card_content = f"""---
license: apache-2.0
tags:
- merge
- evolutionary
- Darwin-A2AP
base_model:
- {model1_hf}
- {model2_hf}
---

# {repo_name}

이 모델은 진화적 알고리즘 v3.2을 사용하여 자동으로 병합된 모델입니다.

## 병합 정보
- **기본 모델 1**: {model1_hf}
- **기본 모델 2**: {model2_hf}
- **병합 비율**: {optimal_ratio:.2f}
- **병합 방법**: {merge_method}

## 진화 알고리즘 최적화 정보
- **검증 데이터셋**: Sklearn Digits (8x8 손글씨 숫자)
- **검증 태스크 정확도**: {best_acc:.2%}
- **참고**: 위 정확도는 병합 비율 최적화를 위한 프록시 지표입니다
- **알고리즘 버전**: Enhanced v3.2 with Persistent Storage

## [WARN] 주의사항
- 이 모델의 실제 언어 생성 성능은 별도 평가가 필요합니다
- 위 검증 점수는 LLM 벤치마크 점수가 아닙니다
"""

            
            # Save model card
            with open(os.path.join(tmp_dir, "README.md"), "w", encoding="utf-8") as f:
                f.write(model_card_content)
            
            progress(0.8, desc="Hugging Face Hub에 업로드 중...")
            
            # Upload to Hub
            api = HfApi(token=hf_token)
            repo_url = api.create_repo(
                repo_id=repo_name,
                private=private,
                exist_ok=True
            )
            
            progress(0.9, desc="모델 파일 업로드 중...")
            
            api.upload_folder(
                folder_path=tmp_dir,
                repo_id=repo_name,
                repo_type="model",
                commit_message=f"Upload evolutionary merged model v3.1"
            )
            
            progress(1.0, desc="완료!")
            
            return f"""[OK] 모델이 성공적으로 업로드되었습니다!
 모델 페이지: https://huggingface.co/{repo_name}
 최종 정확도: {best_acc:.2%}
 알고리즘: Enhanced v3.1"""

    except Exception as e:
        import traceback
        error_details = traceback.format_exc()
        return f"[FAIL] 오류 발생: {str(e)}\n\n상세 오류:\n{error_details}"

def start_experiment_async(
    method, pop_size, runs, total_forward_passes,
    store_train_results, use_pre_trained,
    no_crossover, no_splitpoint, no_matchmaker,
    model1_name, model2_name, model1_file, model2_file,
    model1_hf, model2_hf, merge_ratio, merge_method,
    use_mixed_precision, batch_size, use_architecture_matching,
    track_lineage, use_multi_objective,
    obj_accuracy, obj_speed, obj_size, obj_robustness, obj_generalization,
    progress=gr.Progress()
):
    """비동기 실험 시작"""
    
    progress(0.1, desc="실험 준비 중...")
    
    # 피트니스 목표 준비
    fitness_objectives = [FitnessObjective.ACCURACY]
    fitness_weights = {FitnessObjective.ACCURACY: 1.0}
    
    if use_multi_objective:
        if obj_speed:
            fitness_objectives.append(FitnessObjective.SPEED)
            fitness_weights[FitnessObjective.SPEED] = 0.2
        if obj_size:
            fitness_objectives.append(FitnessObjective.SIZE)
            fitness_weights[FitnessObjective.SIZE] = 0.15
        if obj_robustness:
            fitness_objectives.append(FitnessObjective.ROBUSTNESS)
            fitness_weights[FitnessObjective.ROBUSTNESS] = 0.25
        if obj_generalization:
            fitness_objectives.append(FitnessObjective.GENERALIZATION)
            fitness_weights[FitnessObjective.GENERALIZATION] = 0.25
    
    # 가중치 정규화
    total = sum(fitness_weights.values())
    for key in fitness_weights:
        fitness_weights[key] /= total
    
    # 설정 준비
    config = {
        'method': method,
        'pop_size': pop_size,
        'runs': runs,
        'total_forward_passes': total_forward_passes,
        'store_train_results': store_train_results,
        'use_pre_trained': use_pre_trained,
        'no_crossover': no_crossover,
        'no_splitpoint': no_splitpoint,
        'no_matchmaker': no_matchmaker,
        'model1_name': model1_name,
        'model2_name': model2_name,
        'model1_file': model1_file,
        'model2_file': model2_file,
        'model1_hf': model1_hf,
        'model2_hf': model2_hf,
        'merge_ratio': merge_ratio,
        'merge_method': merge_method,
        'use_mixed_precision': use_mixed_precision,
        'batch_size': batch_size,
        'use_architecture_matching': use_architecture_matching,
        'fitness_objectives': fitness_objectives,
        'fitness_weights': fitness_weights,
        'track_lineage': track_lineage
    }
    
    progress(0.3, desc="실험 등록 중...")
    
    # 실험 생성 및 시작
    exp_id = experiment_manager.create_experiment(config)
    experiment_manager.start_experiment(exp_id, config)
    
    progress(1.0, desc="실험 시작 완료!")
    
    return f"""
[OK] 실험이 백그라운드에서 시작되었습니다!

 **실험 ID: {exp_id}**

이 ID를 저장해두세요. 실험 모니터링 탭에서 언제든지 진행상황을 확인할 수 있습니다.

 모니터링 방법:
1. '실험 모니터링' 탭으로 이동
2. 실험 ID 입력 또는 목록에서 선택
3. 실시간 진행상황 확인

 알고리즘: {method.upper()}
 총 진화 단계: {total_forward_passes:,}
 개체군 크기: {pop_size}
 실행 횟수: {runs}

실험이 완료되면 결과를 다운로드할 수 있습니다.
"""

def check_experiment_status(exp_id):
    """실험 상태 확인"""
    if not exp_id:
        return "실험 ID를 입력하세요.", None, None, None
    
    status = experiment_manager.get_experiment_status(exp_id)
    
    if not status:
        return f"실험 ID '{exp_id}'를 찾을 수 없습니다.", None, None, None
    
    # 부모 모델 정보 가져오기
    exp_details = experiment_manager.get_experiment_with_models(exp_id)
    
    # 상태 텍스트 생성
    status_text = f"""
 실험 상태: {status['status'].upper()}
 알고리즘: {status['method'].upper()}
 진행률: {status['progress']:.1%} ({status['current_step']:,}/{status['total_steps']:,})
 최고 정확도: {status['best_accuracy']:.4f}
 마지막 업데이트: {status['updated_at']}
"""
    
    # 부모 모델 정보 추가
    if exp_details:
        status_text += "\n--- 모델 정보 ---\n"
        if exp_details.get('parent_model1'):
            status_text += f" 모델 1: {exp_details['parent_model1']}\n"
        if exp_details.get('parent_model2'):
            status_text += f" 모델 2: {exp_details['parent_model2']}\n"
        if exp_details.get('merge_ratio'):
            status_text += f" 병합 비율: {exp_details['merge_ratio']}\n"
        if exp_details.get('merge_method'):
            status_text += f" 병합 방법: {exp_details['merge_method']}\n"
    
    if status['error_message']:
        status_text += f"\n[FAIL] 오류: {status['error_message']}"
    

    # 로그 데이터 가져오기
    logs_df = experiment_manager.get_experiment_logs(exp_id, limit=1000)
    
    # 그래프 생성
    fig = None
    if not logs_df.empty:
        fig, axes = plt.subplots(1, 2, figsize=(12, 5))
        
        # 정확도 그래프
        ax = axes[0]
        ax.plot(logs_df['step'], logs_df['train_acc'], label='Train', alpha=0.7)
        ax.plot(logs_df['step'], logs_df['test_acc'], label='Test', alpha=0.7)
        ax.set_xlabel('Steps')
        ax.set_ylabel('Accuracy')
        ax.set_title('Training Progress')
        ax.legend()
        ax.grid(True, alpha=0.3)
        
        # 피트니스 그래프
        ax = axes[1]
        ax.plot(logs_df['step'], logs_df['fitness'], color='green', alpha=0.7)
        ax.set_xlabel('Steps')
        ax.set_ylabel('Fitness')
        ax.set_title('Fitness Evolution')
        ax.grid(True, alpha=0.3)
        
        plt.tight_layout()
    
    # 진행률 바
    progress_value = status['progress']
    
    return status_text, fig, logs_df.head(20) if not logs_df.empty else None, progress_value

def auto_refresh_wrapper(exp_id, enable_refresh):
    """자동 새로고침 래퍼"""
    import time
    
    while enable_refresh and exp_id:
        status_text, fig, logs_df, progress = check_experiment_status(exp_id)
        
        yield status_text, fig, logs_df, progress
        
        # 완료 또는 오류 상태면 중지
        if status_text and ('completed' in status_text.lower() or 'error' in status_text.lower()):
            break
        
        time.sleep(5)  # 5초 대기

def get_all_experiments_display(status_filter="all"):
    """모든 실험 목록 표시"""
    conn = experiment_manager.db.get_connection()
    
    base_query = """
        SELECT id, method, status, 
               CAST(current_step AS REAL) / CAST(total_steps AS REAL) as progress,
               best_accuracy, final_accuracy, 
               model1_hf, model2_hf,
               created_at
        FROM experiments
        {where}
        ORDER BY created_at DESC
        LIMIT 100
    """
    if status_filter == "all":
        query = base_query.format(where="")
        df = pd.read_sql_query(query, conn)
    else:
        query = base_query.format(where="WHERE status = ?")
        df = pd.read_sql_query(query, conn, params=(status_filter,))
    
    conn.close()
    
    if not df.empty:
        # 포맷팅
        df['progress'] = df['progress'].apply(lambda x: f"{x:.1%}" if pd.notna(x) else "0%")
        df['best_accuracy'] = df['best_accuracy'].apply(lambda x: f"{x:.4f}" if pd.notna(x) else "-")
        df['final_accuracy'] = df['final_accuracy'].apply(lambda x: f"{x:.4f}" if pd.notna(x) else "-")
        
        # 모델 정보 간략화
        df['models'] = df.apply(lambda row: 
            f"{row['model1_hf'].split('/')[-1] if row['model1_hf'] else 'N/A'} + " +
            f"{row['model2_hf'].split('/')[-1] if row['model2_hf'] else 'N/A'}", 
            axis=1
        )
        
        # 상태별 이모지 추가
        status_emoji = {
            'completed': '[OK]',
            'running': '', 
            'error': '[FAIL]',
            'pending': ''
        }
        df['status'] = df['status'].apply(lambda x: f"{status_emoji.get(x, '')} {x}")
        
        return df[['id', 'method', 'status', 'progress', 'best_accuracy', 'final_accuracy', 'models', 'created_at']]
    
    return pd.DataFrame({"메시지": ["저장된 실험이 없습니다."]})

    def start_darwin_v4(
        model1, model2, hf_token, merge_method_v4, benchmark,
        upload_repo, pop_size_v4, p1_steps, p2_steps, p2_topk,
        progress=gr.Progress()
    ):
        """Darwin v4 mergekit 실험 시작"""
        if not MERGEKIT_INTEGRATION_AVAILABLE:
            return "[FAIL] mergekit_integration.py 모듈이 없습니다. 같은 디렉토리에 파일이 있는지 확인하세요."

        if not model1.strip() or not model2.strip():
            return "[FAIL] 모델 A와 모델 B를 모두 입력하세요."

        progress(0.2, desc="Darwin v4 실험 등록 중...")

        # 총 스텝 = phase1 + phase2
        total_steps = int(p1_steps) + int(p2_steps)

        config = {
            'method': 'darwin_mergekit',
            'model1_hf': model1.strip(),
            'model2_hf': model2.strip(),
            'hf_token': hf_token.strip() if hf_token else None,
            'merge_method': merge_method_v4,
            'benchmark_preset': benchmark,
            'auto_upload_repo': upload_repo.strip() if upload_repo else '',
            'pop_size': int(pop_size_v4),
            'phase1_steps': int(p1_steps),
            'phase2_steps': int(p2_steps),
            'phase2_top_k': int(p2_topk),
            'total_forward_passes': total_steps,
            'runs': 1,
            'store_train_results': True,
            'fitness_objectives': [FitnessObjective.ACCURACY],
            'fitness_weights': {FitnessObjective.ACCURACY: 1.0},
        }

        exp_id = experiment_manager.create_experiment(config)
        experiment_manager.start_experiment(exp_id, config)

        progress(1.0, desc="Darwin v4 시작!")

        mk_status = "[OK] mergekit 설치됨" if MERGEKIT_AVAILABLE else "[WARN] mergekit 미설치 (pip install mergekit)"
        lm_status = "[OK] lm-eval 설치됨" if LM_EVAL_AVAILABLE else "[WARN] lm-eval 미설치 (휴리스틱 점수 사용)"

        return f"""
[OK] Darwin v4 실험이 백그라운드에서 시작됐습니다!

 실험 ID: {exp_id}

이 ID를 '실험 모니터링' 탭에서 입력하면 실시간 진행상황을 확인할 수 있습니다.

 설정 요약:
  모델 A: {model1.strip()}
  모델 B: {model2.strip()}
  병합 방법: {merge_method_v4}
  벤치마크: {benchmark}
  Phase 1: {p1_steps}스텝 (프록시 탐색)
  Phase 2: {p2_steps}스텝 (실제 mergekit 진화)

 환경:
  {mk_status}
  {lm_status}
{'  결과 자동 업로드: ' + upload_repo if upload_repo else '  자동 업로드: 비활성화'}

 진화 유전자 차원:
  [global_ratio, attn_ratio, ffn_ratio, embed_ratio, density_A, density_B]
"""

_HF_CACHE_AVAILABLE = False

try:

print(f"[MRI] 캐시: {'HF Hub + 로컬' if _HF_CACHE_AVAILABLE else '로컬만'} ({_MRI_CACHE_DIR})")

def _cache_key(model_id):
    return _MRI_CACHE_DIR / (model_id.replace("/", "__").replace("\\", "__") + ".json")

def _hub_path(model_id):
    return "scans/" + model_id.replace("/", "__") + ".json"

def _load_cache(model_id):
    """캐시 로드: 로컬  HF Hub 순서"""
    # 1. 로컬 캐시
    path = _cache_key(model_id)
    if path.exists():
        try:
            with open(path, "r") as f:
                data = json.load(f)
            print(f"[MRI] 로컬 캐시 히트: {model_id}")
            return data
        except Exception:
            pass

    # 2. HF Hub
    if _HF_CACHE_AVAILABLE:
        try:
            local = hf_hub_download(
                repo_id=_HF_MRI_REPO, filename=_hub_path(model_id),
                repo_type="dataset", token=_HF_TOKEN
            )
            with open(local, "r") as f:
                data = json.load(f)
            # 로컬에도 캐시
            with open(path, "w") as f:
                json.dump(data, f, ensure_ascii=False)
            print(f"[MRI] HF Hub 캐시 히트: {model_id}  로컬 복사 완료")
            return data
        except Exception as e:
            print(f"[MRI] HF Hub 로드 실패: {e}")

    return None

def _save_cache(model_id, result):
    """캐시 저장: 로컬 + HF Hub 동시"""
    # 1. 로컬 저장
    try:
        path = _cache_key(model_id)
        with open(path, "w") as f:
            json.dump(result, f, ensure_ascii=False)
        print(f"[MRI] 로컬 저장: {model_id}")
    except Exception as e:
        print(f"[MRI] 로컬 저장 실패: {e}")
        return

    # 2. HF Hub 업로드 (백그라운드)
    if _HF_CACHE_AVAILABLE:
        def _upload():
            try:
                upload_file(
                    path_or_fileobj=str(path),
                    path_in_repo=_hub_path(model_id),
                    repo_id=_HF_MRI_REPO,
                    repo_type="dataset",
                    token=_HF_TOKEN,
                    commit_message=f"MRI scan: {model_id}"
                )
                print(f"[MRI] HF Hub 업로드 완료: {model_id}")
            except Exception as e:
                print(f"[MRI] HF Hub 업로드 실패: {e}")
        threading.Thread(target=_upload, daemon=True).start()

def _list_hub_scans():
    """HF Hub에 저장된 전체 스캔 목록"""
    results = []
    # 로컬
    for f in _MRI_CACHE_DIR.glob("*.json"):
        results.append({"model_id": f.stem.replace("__", "/"), "source": "local", "file": f.name, "size": f.stat().st_size})
    # HF Hub
    if _HF_CACHE_AVAILABLE:
        try:
            files = _hf_api.list_repo_files(_HF_MRI_REPO, repo_type="dataset")
            for f in files:
                if f.startswith("scans/") and f.endswith(".json"):
                    mid = f.replace("scans/", "").replace(".json", "").replace("__", "/")
                    if not any(r["model_id"] == mid for r in results):
                        results.append({"model_id": mid, "source": "hub", "file": f})
        except Exception:
            pass
    return results

def _run_scan_bg(job_id, model_id):
    """백그라운드 스캔 스레드 — 캐시 우선"""
    try:
        # ── 캐시 체크 ──
        cached = _load_cache(model_id)
        if cached:
            _mri_jobs[job_id]["result"] = cached
            _mri_jobs[job_id]["status"] = "done"
            _mri_jobs[job_id]["progress"] = " 캐시에서 로드 (즉시)"
            return

        # ── 실제 스캔 ──
        _mri_jobs[job_id]["progress"] = "모델 다운로드 중..."
        from mri_bridge import MRIQuickScanner
        from transformers import AutoModelForCausalLM, AutoTokenizer
        tokenizer = AutoTokenizer.from_pretrained(model_id, trust_remote_code=True)
        if tokenizer.pad_token is None:
            tokenizer.pad_token = tokenizer.eos_token
        _mri_jobs[job_id]["progress"] = "가중치 로딩 중..."
        model = AutoModelForCausalLM.from_pretrained(
            model_id, torch_dtype=torch.float16, device_map="auto",
            trust_remote_code=True, low_cpu_mem_usage=True)
        model.eval()
        _mri_jobs[job_id]["progress"] = "MRI 스캔 중... (수 분 소요)"
        scanner = MRIQuickScanner(model, tokenizer)
        scan = scanner.quick_scan()

        # numpy  Python 네이티브 변환
        def to_native(obj):
            import numpy as np
            if isinstance(obj, dict):
                return {k: to_native(v) for k, v in obj.items()}
            elif isinstance(obj, (list, tuple)):
                return [to_native(v) for v in obj]
            elif isinstance(obj, (np.integer,)):
                return int(obj)
            elif isinstance(obj, (np.floating,)):
                return float(obj)
            elif isinstance(obj, np.ndarray):
                return obj.tolist()
            elif isinstance(obj, (np.bool_,)):
                return bool(obj)
            return obj

        moe_details = [{
            "layer_idx": int(d.layer_idx), "num_experts": int(d.num_experts),
            "router_entropy": float(d.router_entropy),
            "expert_utilization": [float(x) for x in d.expert_utilization],
            "dead_experts": [int(x) for x in d.dead_experts],
            "expert_importance": [float(x) for x in d.expert_importance],
            "expert_similarity_avg": float(d.expert_similarity_avg),
        } for d in scan.moe_layer_details]

        layer_imp = {}
        for k, v in scan.layer_importance.items():
            layer_imp[str(k)] = [float(x) for x in v]

        overall = {}
        for k, v in scan.overall_scores.items():
            overall[str(k)] = float(v)

        result = {
            "model_id": str(scan.model_id or model_id),
            "n_layers": int(scan.n_layers),
            "arch_type": str(scan.arch_type),
            "layer_importance": layer_imp,
            "overall_scores": overall,
            "scan_time": float(scan.scan_time),
            "moe_info": to_native(scan.moe_info) if scan.moe_info else None,
            "moe_layer_details": moe_details,
        }

        # JSON 직렬화 테스트
        import json
        json.dumps(result)

        # ── 캐시 저장 ──
        _save_cache(model_id, result)

        _mri_jobs[job_id]["result"] = result
        _mri_jobs[job_id]["status"] = "done"
        _mri_jobs[job_id]["progress"] = "완료!"
        print(f"[MRI] 스캔 완료: {model_id} ({scan.n_layers}L, {scan.scan_time:.1f}s)")
        del model; torch.cuda.empty_cache()
    except Exception as e:
        import traceback
        traceback.print_exc()
        _mri_jobs[job_id]["status"] = "error"
        _mri_jobs[job_id]["progress"] = f"오류: {e}"

@_fastapi_app.post("/api/mri/scan")
async def api_mri_scan(request: Request):
    """스캔 시작  job_id 즉시 반환 (프록시 타임아웃 우회)"""
    try:
        body = await request.json()
        model_id = body.get("model_id", "")
        if not model_id:
            return JSONResponse({"error": "model_id required"}, status_code=400)
        if not MRI_BRIDGE_AVAILABLE:
            return JSONResponse({"error": "MRI Bridge not available"}, status_code=503)
        job_id = str(uuid.uuid4())[:8]
        _mri_jobs[job_id] = {"status": "running", "progress": "시작...", "result": None}
        t = threading.Thread(target=_run_scan_bg, args=(job_id, model_id), daemon=True)
        t.start()
        return JSONResponse({"job_id": job_id, "status": "running"})
    except Exception as e:
        return JSONResponse({"error": str(e)}, status_code=500)

@_fastapi_app.get("/api/mri/status/{job_id}")
async def api_mri_status(job_id: str):
    """스캔 진행 상태 폴링"""
    job = _mri_jobs.get(job_id)
    if not job:
        return JSONResponse({"error": "job not found"}, status_code=404)
    resp = {"status": job["status"], "progress": job["progress"]}
    if job["status"] == "done" and job["result"]:
        resp["result"] = job["result"]
    return JSONResponse(resp)

@_fastapi_app.get("/api/mri/cache/load")

@_fastapi_app.post("/api/darwin/upload/{exp_id}")
async def api_darwin_upload(exp_id: str, request: Request):
    """Darwin 실험 결과를 HF Hub에 업로드"""
    try:
        body = await request.json()
        repo_id = body.get('repo_id', '')
        hf_token = body.get('hf_token', os.environ.get('HF_TOKEN', ''))

        # 실험 상태에서 best_merge_path 가져오기
        status = experiment_manager.get_experiment_status(exp_id)
        if not status:
            return JSONResponse({"error": "experiment not found"}, status_code=404)

        best_merge_path = status.get('best_merge_path', '')
        best_score = status.get('best_score', 0)
        best_genome = status.get('best_genome', [])

        if not best_merge_path or not Path(best_merge_path).exists():
            return JSONResponse({
                "error": "병합 모델 경로가 없거나 삭제되었습니다. (휘발성 스토리지 재시작으로 손실 가능)",
                "best_merge_path": best_merge_path
            }, status_code=404)

        # repo_id 자동 생성
        if not repo_id:
            try:
                from huggingface_hub import HfApi as _HfApi
                _api = _HfApi(token=hf_token)
                _owner = _api.whoami().get('name', 'user')
            except Exception:
                _owner = 'user'
            repo_id = f"{_owner}/Darwin-{exp_id[:8]}-score{best_score:.4f}".replace('.', '')

        import numpy as np
        genome = np.array(best_genome) if best_genome else None
        model1_id = status.get('model1', status.get('config', {}).get('model1_hf', ''))
        model2_id = status.get('model2', status.get('config', {}).get('model2_hf', ''))

        ok, msg = upload_darwin_result_to_hf(
            merged_model_path=best_merge_path,
            repo_id=repo_id,
            genome=genome,
            score=best_score,
            model1_id=model1_id,
            model2_id=model2_id,
            hf_token=hf_token,
        )
        print(f"[Upload] {msg}")
        return JSONResponse({"success": ok, "message": msg, "repo_id": repo_id})

    except Exception as e:
        import traceback; traceback.print_exc()
        return JSONResponse({"error": str(e)}, status_code=500)
