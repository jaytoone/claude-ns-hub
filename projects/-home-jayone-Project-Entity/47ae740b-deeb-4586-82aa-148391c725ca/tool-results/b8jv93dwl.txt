#!/bin/bash
# zellij CSK 멀티 세션 관리 스크립트 (celi)
# cmux(tmux-csk-multi.sh)의 zellij 버전
# 작성일: 2026-02-26

ZELLIJ="${HOME}/.local/bin/zellij"
FZF="${HOME}/.fzf/bin/fzf"
CONFIG_FILE="${HOME}/.config/tmux-csk-sessions.conf"
POS_MAP="${HOME}/.config/celi-pos-map.conf"  # 세션명 → config 순서 매핑
SESSION_PREFIX=""

if [ -z "$NO_COLOR" ]; then
    RED=$'\033[0;31m'
    GREEN=$'\033[0;32m'
    YELLOW=$'\033[1;33m'
    BLUE=$'\033[0;34m'
    CYAN=$'\033[0;36m'
    MAGENTA=$'\033[0;35m'
    NC=$'\033[0m'
else
    RED=''; GREEN=''; YELLOW=''; BLUE=''; CYAN=''; MAGENTA=''; NC=''
fi

declare -a SESS_PROJECT SESS_NAME SESS_DIR
declare -A SNAME_MAP  # "project:session" → "NN-project-session"

# ============================================================================
# 공통 유틸
# ============================================================================

# load_sessions 후: SNAME_MAP에서 숫자 prefix 포함 이름 반환
# load_sessions 전(직접 호출): prefix 없는 기본 이름 반환
get_session_name() {
    local key="${1}:${2}"
    if [ -n "${SNAME_MAP[$key]+_}" ]; then
        echo "${SNAME_MAP[$key]}"
    else
        echo "${1}-${2}"
    fi
}

# zellij 세션 존재 확인 (ANSI 제거 + EXITED 상태 제외)
session_exists() {
    "$ZELLIJ" list-sessions 2>/dev/null \
        | sed $'s/\033\\[[0-9;]*m//g' \
        | grep -iv 'exited' \
        | awk '{print $1}' \
        | grep -qx "$1"
}

# 실행 중인 csk 세션 목록 (EXITED 제외, 영숫자로 시작하는 세션명만)
list_zellij_sessions() {
    "$ZELLIJ" list-sessions 2>/dev/null \
        | sed $'s/\033\\[[0-9;]*m//g' \
        | grep -iv 'exited' \
        | awk '{print $1}' \
        | grep -E '^[A-Za-z0-9]'
}

# config 순서 기준 pos-map 생성 (세션명 → config 위치, hooks 팝업 번호용)
# 형식: PROJECT-SESSION=NN  (예: MONET-Broadcaster-main=11)
write_pos_map() {
    mkdir -p "$(dirname "$POS_MAP")"
    local tmpfile="${POS_MAP}.tmp"
    echo "# celi pos-map: SESSION_NAME=config_position" > "$tmpfile"
    local pos=0
    while IFS= read -r line || [ -n "$line" ]; do
        line=$(echo "$line" | xargs 2>/dev/null)
        [[ -z "$line" || "$line" =~ ^# ]] && continue
        [[ ! "$line" =~ ^[^:]+:[^:]+:.+ ]] && continue
        IFS=':' read -r p s d <<< "$line"
        p=$(echo "$p"|xargs); s=$(echo "$s"|xargs)
        pos=$((pos+1))
        echo "${p}-${s}=$(printf '%02d' $pos)" >> "$tmpfile"
    done < "$CONFIG_FILE"
    mv -f "$tmpfile" "$POS_MAP"
}

# 설정 파일 파서
# config 형식: PROJECT:SESSION:DIR
load_sessions() {
    SESS_PROJECT=(); SESS_NAME=(); SESS_DIR=(); SNAME_MAP=()
    if [ ! -f "$CONFIG_FILE" ]; then
        echo -e "${RED}설정 파일 없음: ${CONFIG_FILE}${NC}" >&2
        return 1
    fi
    local n=0 ok=0
    while IFS= read -r line || [ -n "$line" ]; do
        n=$((n+1))
        line=$(echo "$line" | xargs 2>/dev/null)
        [[ -z "$line" || "$line" =~ ^# ]] && continue
        if [[ ! "$line" =~ ^[^:]+:[^:]+:.+ ]]; then
            echo -e "${YELLOW}라인 ${n}: 형식 오류 (스킵)${NC}" >&2; continue
        fi
        IFS=':' read -r p s d <<< "$line"
        p=$(echo "$p"|xargs); s=$(echo "$s"|xargs); d=$(echo "$d"|xargs)
        if [ ! -d "$d" ]; then
            echo -e "${YELLOW}라인 ${n}: 디렉토리 없음 → ${d} (스킵)${NC}" >&2; continue
        fi
        ok=$((ok+1))
        SESS_PROJECT+=("$p"); SESS_NAME+=("$s"); SESS_DIR+=("$d")
        SNAME_MAP["$p:$s"]="${p}-${s}"
    done < "$CONFIG_FILE"
    [ $ok -eq 0 ] && echo -e "${RED}유효한 세션 없음${NC}" >&2 && return 1
    write_pos_map
    return 0
}

# ============================================================================
# 세션 생성
# ============================================================================

create_session() {
    local project="$1" session="$2" directory="$3"
    local sname
    sname=$(get_session_name "$project" "$session")

    if session_exists "$sname"; then
        echo -e "${YELLOW}[${project}:${session}] 이미 실행 중 (스킵)${NC}"
        return 1
    fi

    echo -e "${GREEN}[${project}:${session}] 세션 생성 중...${NC}"
    echo -e "  → ${directory}"
    echo -e "  → ${CYAN}${sname}${NC}"

    # csk 자동 실행용 임시 셸 래퍼 (SHELL 환경변수로 등록)
    local tmpsh flagfile
    tmpsh=$(mktemp /tmp/.csk-init-XXXXXX.sh)
    flagfile="/tmp/.csk-ran-${sname}"
    rm -f "$flagfile"  # 재시작 시 이전 플래그 제거

    cat > "$tmpsh" << SHELL_EOF
#!/bin/bash
shopt -s expand_aliases
[ -f "\${HOME}/.bashrc" ] && source "\${HOME}/.bashrc"
[ -f "\${HOME}/.bash_aliases" ] && source "\${HOME}/.bash_aliases"
cd "${directory}"
# 첫 pane에서만 csk 실행 (이후 새 pane은 bash로 바로 진입)
if [ ! -f "${flagfile}" ]; then
    touch "${flagfile}"
    csk
fi
exec bash
SHELL_EOF
    chmod +x "$tmpsh"

    # 현재 터미널 크기 읽기 (attach 시 화면 크기 불일치 방지)
    local _cols _rows
    _cols=$(tput cols 2>/dev/null || echo 220)
    _rows=$(tput lines 2>/dev/null || echo 50)

    # script fake TTY + SHELL 래퍼로 세션 생성
    # stty cols/rows: 백그라운드 PTY를 현재 터미널과 같은 크기로 초기화
    script -q -c "bash -c 'stty cols ${_cols} rows ${_rows} 2>/dev/null; SHELL=${tmpsh} ${ZELLIJ} -s ${sname}'" /dev/null < /dev/null > /dev/null 2>&1 &

    # TTY 설정 복구 (script가 터미널 상태를 변경할 수 있으므로 방어)
    stty sane 2>/dev/null || true

    # 생성 완료 대기 (폴링: 최대 3초, 0.2s 간격)
    local waited=0
    while [ $waited -lt 15 ]; do
        sleep 0.2; waited=$((waited+1))
        session_exists "$sname" && break
    done

    if session_exists "$sname"; then
        echo -e "${GREEN}  생성 완료${NC}"
        return 0
    else
        rm -f "$tmpsh"
        echo -e "${RED}  생성 실패${NC}"
        return 2
    fi
}

# create_session에서 백그라운드 실행만 담당 (start_all_sessions 병렬용)
_launch_session_bg() {
    local project="$1" session="$2" directory="$3"
    local sname
    sname=$(get_session_name "$project" "$session")

    session_exists "$sname" && return 1

    local tmpsh flagfile
    tmpsh=$(mktemp /tmp/.csk-init-XXXXXX.sh)
    flagfile="/tmp/.csk-ran-${sname}"
    rm -f "$flagfile"

    cat > "$tmpsh" << SHELL_EOF
#!/bin/bash
shopt -s expand_aliases
[ -f "\${HOME}/.bashrc" ] && source "\${HOME}/.bashrc"
[ -f "\${HOME}/.bash_aliases" ] && source "\${HOME}/.bash_aliases"
cd "${directory}"
if [ ! -f "${flagfile}" ]; then
    touch "${flagfile}"
    csk
fi
exec bash
SHELL_EOF
    chmod +x "$tmpsh"

    local _cols _rows
    _cols=$(tput cols 2>/dev/null || echo 220)
    _rows=$(tput lines 2>/dev/null || echo 50)

    script -q -c "bash -c 'stty cols ${_cols} rows ${_rows} 2>/dev/null; SHELL=${tmpsh} ${ZELLIJ} -s ${sname}'" /dev/null < /dev/null > /dev/null 2>&1 &
    return 0
}

# ============================================================================
# start
# ============================================================================

start_all_sessions() {
    load_sessions || return 1
    local total=${#SESS_PROJECT[@]} success=0 skipped=0 failed=0

    echo -e "${BLUE}zellij CSK 멀티 세션 시작...${NC}"
    echo -e "${CYAN}설정 파일: ${CONFIG_FILE}${NC}"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

    # Phase 0: EXITED 상태 세션 삭제 (이름 점유로 새 세션 생성 막히는 문제 방지)
    local raw_sessions
    raw_sessions=$("$ZELLIJ" list-sessions 2>/dev/null | sed $'s/\033\\[[0-9;]*m//g')
    for i in $(seq 0 $((total-1))); do
        local sname_pre
        sname_pre=$(get_session_name "${SESS_PROJECT[$i]}" "${SESS_NAME[$i]}")
        if echo "$raw_sessions" | grep "^${sname_pre}" | grep -qi 'exited'; then
            "$ZELLIJ" delete-session "$sname_pre" 2>/dev/null
        fi
    done

    # Phase 1: 0.5s 간격으로 stagger 시작 (동시 시작 시 zellij 충돌 방지)
    local -a pending=()
    for i in $(seq 0 $((total-1))); do
        local p="${SESS_PROJECT[$i]}" s="${SESS_NAME[$i]}" d="${SESS_DIR[$i]}"
        local sname
        sname=$(get_session_name "$p" "$s")
        if session_exists "$sname"; then
            echo -e "${YELLOW}[${p}:${s}] 이미 실행 중 (스킵)${NC}"
            skipped=$((skipped+1))
        else
            echo -e "${GREEN}[${p}:${s}]${NC} 시작 → ${CYAN}${sname}${NC}"
            _launch_session_bg "$p" "$s" "$d"
            pending+=("$sname")
            sleep 3  # zellij 소켓 초기화 충돌 방지 (CPU 과부하 방지용 stagger)
        fi
    done
    stty sane 2>/dev/null || true

    # Phase 2: 모든 세션 생성 확인 (최대 10초 일괄 대기)
    if [ ${#pending[@]} -gt 0 ]; then
        echo ""
        echo -n "  확인 중"
        local waited=0
        while [ $waited -lt 50 ]; do
            sleep 0.2; waited=$((waited+1))
            echo -n "."
            local all_done=1
            for sname in "${pending[@]}"; do
                session_exists "$sname" || { all_done=0; break; }
            done
            [ $all_done -eq 1 ] && break
        done
        echo ""

        for sname in "${pending[@]}"; do
            if session_exists "$sname"; then
                echo -e "  ${GREEN}[완료]${NC} ${sname}"
                success=$((success+1))
            else
                echo -e "  ${RED}[실패]${NC} ${sname}"
                failed=$((failed+1))
            fi
        done
    fi

    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo -e "${GREEN}완료: ${success}개 생성${NC}"
    [ $skipped -gt 0 ] && echo -e "${YELLOW}스킵: ${skipped}개${NC}"
    [ $failed -gt 0 ] && echo -e "${RED}실패: ${failed}개${NC}"
    echo ""
    echo -e "${YELLOW}세션 목록: celi list${NC}"
    echo -e "${YELLOW}세션 접속: celi attach <프로젝트> <세션>${NC}"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
}

# ============================================================================
# list
# ============================================================================

list_sessions() {
    local target_project="${1:-}"
    load_sessions || return 1

    local total=${#SESS_PROJECT[@]} running=0 stopped=0
    local current_project="" project_count=0
    local pos=0  # config 파일 기준 순서 (항상 1,2,3... 유지)

    if [ -n "$target_project" ]; then
        echo -e "${BLUE}프로젝트 세션: ${MAGENTA}${target_project}${NC}"
    else
        echo -e "${BLUE}zellij CSK 세션 목록${NC}"
    fi
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo ""

    for i in $(seq 0 $((total-1))); do
        local p="${SESS_PROJECT[$i]}" s="${SESS_NAME[$i]}" d="${SESS_DIR[$i]}"
        [ -n "$target_project" ] && [ "$p" != "$target_project" ] && continue
        local sname
        sname=$(get_session_name "$p" "$s")
        pos=$((pos+1))
        local pos_label
        pos_label=$(printf '%02d' $pos)

        if [ "$p" != "$current_project" ]; then
            [ -n "$current_project" ] && echo ""
            project_count=$(printf '%s\n' "${SESS_PROJECT[@]}" | grep -c "^${p}$")
            echo -e "${MAGENTA}${p}${NC} ${YELLOW}(${project_count}개 세션)${NC}"
            current_project="$p"
        fi

        if session_exists "$sname"; then
            echo -e "  ${GREEN}[실행중]${NC} #${pos_label}  ${p}:${s}"
            running=$((running+1))
        else
            echo -e "  ${RED}[중단됨]${NC} #${pos_label}  ${p}:${s}"
            stopped=$((stopped+1))
        fi
        echo -e "            ${d}"
    done

    echo ""
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo -e "${GREEN}실행 중: ${running}개${NC} | ${RED}중단: ${stopped}개${NC} | 전체: $((running+stopped))개"
    echo ""
    echo -e "${YELLOW}접속: celi attach <프로젝트> <세션>${NC}"
    echo -e "${YELLOW}종료: celi stop [프로젝트] [세션]${NC}"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
}

# ============================================================================
# attach
# ============================================================================

attach_session() {
    local p="$1" s="$2"
    if [ -z "$p" ] || [ -z "$s" ]; then
        echo -e "${RED}사용법: celi attach <프로젝트> <세션>${NC}"
        return 1
    fi
    local sname
    sname=$(get_session_name "$p" "$s")
    if ! session_exists "$sname"; then
        echo -e "${RED}세션 없음: ${sname}${NC}"
        echo -e "${YELLOW}세션 시작: celi start${NC}"
        return 1
    fi
    echo -e "${GREEN}세션 접속: ${CYAN}${sname}${NC}"
    "$ZELLIJ" attach "$sname"
}

# ============================================================================
# stop
# ============================================================================

_kill_session() {
    local sname="$1"
    echo -e "${YELLOW}종료: ${CYAN}${sname}${NC}"
    "$ZELLIJ" kill-session "$sname" 2>/dev/null
    sleep 0.3
    # EXITED 상태로 남는 세션을 완전 제거 (zellij의 resurrect 기능 때문)
    "$ZELLIJ" delete-session "$sname" 2>/dev/null
    rm -f "/tmp/.csk-ran-${sname}" 2>/dev/null
    echo -e "${GREEN}  완료${NC}"
}

stop_session() {
    local target_p="${1:-}" target_s="${2:-}"

    if [ -z "$target_p" ]; then
        # 전체 종료
        load_sessions || return 1
        local running=0
        for i in "${!SESS_PROJECT[@]}"; do
            local sname
            sname=$(get_session_name "${SESS_PROJECT[$i]}" "${SESS_NAME[$i]}")
            session_exists "$sname" && running=$((running+1))
        done
        [ $running -eq 0 ] && echo -e "${YELLOW}실행 중인 세션 없음${NC}" && return 0
        echo -e "${YELLOW}${running}개 세션 종료합니다.${NC}"
        read -p "계속하시겠습니까? (y/N): " c
        [[ ! "$c" =~ ^[Yy]$ ]] && echo "취소" && return 0
        for i in "${!SESS_PROJECT[@]}"; do
            local sname
            sname=$(get_session_name "${SESS_PROJECT[$i]}" "${SESS_NAME[$i]}")
            session_exists "$sname" && _kill_session "$sname"
        done
        return 0
    fi

    if [ -z "$target_s" ]; then
        # 프로젝트 전체 종료
        load_sessions || return 1
        local found=0
        for i in "${!SESS_PROJECT[@]}"; do
            [ "${SESS_PROJECT[$i]}" = "$target_p" ] && found=$((found+1))
        done
        [ $found -eq 0 ] && echo -e "${RED}프로젝트 없음: ${target_p}${NC}" && return 1
        echo -e "${YELLOW}${target_p}의 ${found}개 세션 종료합니다.${NC}"
        read -p "계속하시겠습니까? (y/N): " c
        [[ ! "$c" =~ ^[Yy]$ ]] && echo "취소" && return 0
        for i in "${!SESS_PROJECT[@]}"; do
            if [ "${SESS_PROJECT[$i]}" = "$target_p" ]; then
                local sname
                sname=$(get_session_name "$target_p" "${SESS_NAME[$i]}")
                session_exists "$sname" && _kill_session "$sname"
            fi
        done
        return 0
    fi

    # 특정 세션 종료
    local sname
    sname=$(get_session_name "$target_p" "$target_s")
    if ! session_exists "$sname"; then
        echo -e "${RED}세션 없음: ${sname}${NC}" && return 1
    fi
    echo -e "${YELLOW}세션 종료: ${CYAN}${sname}${NC}"
    read -p "계속하시겠습니까? (y/N): " c
    [[ ! "$c" =~ ^[Yy]$ ]] && echo "취소" && return 0
    _kill_session "$sname"
}

# ============================================================================
# restart
# ============================================================================

restart_sessions() {
    local target_p="${1:-}"
    load_sessions || return 1

    echo -e "${BLUE}세션 재시작${NC}"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

    # 종료
    local killed=0
    for i in "${!SESS_PROJECT[@]}"; do
        local p="${SESS_PROJECT[$i]}" s="${SESS_NAME[$i]}"
        [ -n "$target_p" ] && [ "$p" != "$target_p" ] && continue
        local sname
        sname=$(get_session_name "$p" "$s")
        if session_exists "$sname"; then
            _kill_session "$sname"
            killed=$((killed+1))
        fi
    done
    echo -e "${GREEN}${killed}개 세션 종료 완료${NC}"
    echo ""

    # 재시작 (start_all_sessions와 동일한 stagger 방식)
    local -a pending=()
    for i in "${!SESS_PROJECT[@]}"; do
        local p="${SESS_PROJECT[$i]}" s="${SESS_NAME[$i]}" d="${SESS_DIR[$i]}"
        [ -n "$target_p" ] && [ "$p" != "$target_p" ] && continue
        local sname
        sname=$(get_session_name "$p" "$s")
        echo -e "${GREEN}[${p}:${s}]${NC} 시작 → ${CYAN}${sname}${NC}"
        _launch_session_bg "$p" "$s" "$d"
        pending+=("$sname")
        sleep 3
    done
    stty sane 2>/dev/null || true

    # Phase 2: 일괄 대기 (최대 10초)
    if [ ${#pending[@]} -gt 0 ]; then
        echo ""
        echo -n "  확인 중"
        local waited=0
        while [ $waited -lt 50 ]; do
            sleep 0.2; waited=$((waited+1))
            echo -n "."
            local all_done=1
            for sname in "${pending[@]}"; do
                session_exists "$sname" || { all_done=0; break; }
            done
            [ $all_done -eq 1 ] && break
        done
        echo ""

        local success=0 failed=0
        for sname in "${pending[@]}"; do
            if session_exists "$sname"; then
                echo -e "  ${GREEN}[완료]${NC} ${sname}"
                success=$((success+1))
            else
                echo -e "  ${RED}[실패]${NC} ${sname}"
                failed=$((failed+1))
            fi
        done
    fi

    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo -e "${GREEN}${success}개 세션 재시작 완료${NC}"
    [ $failed -gt 0 ] && echo -e "${RED}실패: ${failed}개${NC}"
}

# ============================================================================
# status
# ============================================================================

check_status() {
    load_sessions || return 1
    local total=${#SESS_PROJECT[@]} running=0 stopped=0

    for i in "${!SESS_PROJECT[@]}"; do
        local sname
        sname=$(get_session_name "${SESS_PROJECT[$i]}" "${SESS_NAME[$i]}")
        if session_exists "$sname"; then running=$((running+1)); else stopped=$((stopped+1)); fi
    done

    echo -e "${BLUE}zellij CSK 세션 상태${NC}"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo -e "${GREEN}실행 중: ${running}개${NC}"
    echo -e "${RED}중단됨: ${stopped}개${NC}"
    echo -e "${CYAN}전체: ${total}개${NC}"
    echo ""
    [ $running -gt 0 ] && echo -e "${YELLOW}상세 목록: celi list${NC}" || echo -e "${YELLOW}세션 시작: celi start${NC}"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
}

# ============================================================================

edit_config() {
    [ ! -f "$CONFIG_FILE" ] && echo -e "${RED}설정 파일 없음: ${CONFIG_FILE}${NC}" && return 1
    local ed="${EDITOR:-}"
    if [ -z "$ed" ]; then
        for candidate in nvim vim nano; do
            command -v "$candidate" &>/dev/null && ed="$candidate" && break
        done
    fi
    [ -z "$ed" ] && echo -e "${YELLOW}에디터 없음. 경로: ${CONFIG_FILE}${NC}" && return 1
    "$ed" "$CONFIG_FILE"
}

show_config() {
    echo -e "${BLUE}설정 파일: ${CONFIG_FILE}${NC}"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    [ -f "$CONFIG_FILE" ] && cat -n "$CONFIG_FILE" || echo -e "${YELLOW}설정 파일 없음${NC}"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
}

# ============================================================================
# fzf 인터랙티브 선택기 (인자 없을 때)
# ============================================================================

interactive_picker() {
    local fzf_bin
    fzf_bin=$(command -v fzf 2>/dev/null || echo "$FZF")
    if [[ ! -x "$fzf_bin" ]]; then
        echo -e "${YELLOW}fzf 미설치 — celi list 를 사용하세요${NC}"
        list_sessions; return 0
    fi

    # detach 후 자동으로 fzf 다시 표시 (Ctrl-c / Esc → 빈 선택 → 루프 종료)
    while true; do
        clear
        load_sessions || return 1

        local running_sessions
        running_sessions=$(list_zellij_sessions)

        local all_entries=()
        local fzf_pos=0
        for i in "${!SESS_PROJECT[@]}"; do
            local sname
            sname=$(get_session_name "${SESS_PROJECT[$i]}" "${SESS_NAME[$i]}")
            fzf_pos=$((fzf_pos+1))
            local pos_lbl
            pos_lbl=$(printf '%02d' $fzf_pos)
            local tag
            if echo "$running_sessions" | grep -qx "$sname"; then
                tag="${GREEN}[실행중]${NC}"
            else
                tag="${YELLOW}[중단됨]${NC}"
            fi
            all_entries+=("$(printf '#%s %s %b  %s' "$pos_lbl" "$sname" "$tag" "${SESS_DIR[$i]}")")
        done

        # 설정에 없는 실행 중 csk 세션 추가
        while IFS= read -r s; do
            [[ -z "$s" ]] && continue
            local found=0
            for i in "${!SESS_PROJECT[@]}"; do
                local sn
                sn=$(get_session_name "${SESS_PROJECT[$i]}" "${SESS_NAME[$i]}")
                [ "$sn" = "$s" ] && found=1 && break
            done
            [ $found -eq 0 ] && all_entries+=("$(printf '%s %b' "$s" "${GREEN}[실행중]${NC}")")
        done <<< "$running_sessions"

        [ ${#all_entries[@]} -eq 0 ] && echo -e "${YELLOW}등록된 세션 없음 — celi start 로 시작하세요${NC}" && return 0

        local selected
        selected=$(printf '%b\n' "${all_entries[@]}" | \
            "$fzf_bin" --ansi --reverse \
                --header="[Enter] 세션 전환  [Alt-k] 세션 종료  [Ctrl-c] 종료" \
                --preview="$ZELLIJ list-sessions 2>/dev/null | grep {2} || echo '(중단됨)'" \
                --bind="alt-k:execute($ZELLIJ kill-session {2} 2>/dev/null)+reload(${ZELLIJ} list-sessions 2>/dev/null | sed 's/\x1b\[[0-9;]*m//g' | grep -iv exited | awk '{print \$1}')" \
            | awk '{print $2}')  # $1=#NN, $2=세션명

        [ -z "$selected" ] && return 0  # Ctrl-c / Esc → 종료

        if echo "$running_sessions" | grep -qx "$selected"; then
            echo -e "${GREEN}세션 접속: ${CYAN}${selected}${NC}"
            "$ZELLIJ" attach "$selected"
        else
            echo -e "${GREEN}세션 시작: ${CYAN}${selected}${NC}"
            "$ZELLIJ" -s "$selected"
        fi
        # zellij 종료(detach) 후 루프 재진입 → fzf 다시 표시
    done
}

# ============================================================================
# sync: pos-map 재생성 + 실행 현황 확인
# (세션명에 prefix 없으므로 레지스트리 불필요 — config 순서 = 표시 번호)
# ============================================================================
sync_sessions() {
    load_sessions || return 1

    local active_sessions
    active_sessions=$("$ZELLIJ" list-sessions 2>/dev/null \
        | sed $'s/\033\\[[0-9;]*m//g' \
        | grep -iv exited \
        | awk '{print $1}')

    echo -e "${BLUE}세션 동기화 중...${NC}"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

    local ok=0 not_found=0
    local total=${#SESS_PROJECT[@]}
    for i in $(seq 0 $((total-1))); do
        local p="${SESS_PROJECT[$i]}" s="${SESS_NAME[$i]}"
        local sname="${p}-${s}"
        if echo "$active_sessions" | grep -qx "$sname"; then
            ok=$((ok+1))
        else
            echo -e "${CYAN}미실행:${NC} ${sname}"
            not_found=$((not_found+1))
        fi
    done

    local exited_count
    exited_count=$("$ZELLIJ" list-sessions 2>/dev/null | sed $'s/\033\\[[0-9;]*m//g' | grep -ic exited || echo 0)

    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo -e "${GREEN}실행중: ${ok}개${NC} | ${CYAN}미실행: ${not_found}개${NC}"
    if [ "$exited_count" -gt 0 ]; then
        echo -e "${YELLOW}EXITED 세션 ${exited_count}개 발견 — celi clean 으로 정리하세요${NC}"
    fi
}

# ============================================================================
# clean (EXITED 세션 일괄 삭제)
# ============================================================================

clean_exited_sessions() {
    load_sessions || return 1

    local raw_sessions clean_list=()
    raw_sessions=$("$ZELLIJ" list-sessions 2>/dev/null | sed $'s/\033\\[[0-9;]*m//g')

    # 1) EXITED 상태 세션 수집 (대소문자 무관)
    while IFS= read -r line; do
        [[ -z "$line" ]] && continue
        if echo "$line" | grep -qi 'exited'; then
            local sname
            sname=$(echo "$line" | awk '{print $1}')
            clean_list+=("$sname")
        fi
    done <<< "$raw_sessions"

    # 2) 설정에 없는 고아(orphan) 세션 수집
    #    - ^[0-9][0-9]-  : 현재 형식 (01-APP-...)
    #    - ^csk-         : 이전 prefix 형식 잔여 세션 포함
    #    - ^[A-Z][A-Z]+- : 구형 PROJECT-SESSION[-branch] 형식 (e.g. MONET-PaintPoint-main)
    local running_managed
    running_managed=$(echo "$raw_sessions" | grep -iv 'exited' \
        | awk '{print $1}' \
        | grep -E '^[0-9][0-9]-|^csk-|^[A-Z][A-Z]+-')
    while IFS= read -r sname; do
        [[ -z "$sname" ]] && continue
        local found=0
        for i in "${!SESS_PROJECT[@]}"; do
            local expected
            expected=$(get_session_name "${SESS_PROJECT[$i]}" "${SESS_NAME[$i]}")
            [ "$sname" = "$expected" ] && found=1 && break
        done
        [ $found -eq 0 ] && clean_list+=("$sname")
    done <<< "$running_managed"

    # 중복 제거
    local -A seen; local unique_list=()
    for s in "${clean_list[@]}"; do
        [ -z "${seen[$s]+_}" ] && seen[$s]=1 && unique_list+=("$s")
    done

    if [ ${#unique_list[@]} -eq 0 ]; then
        echo -e "${GREEN}정리할 세션 없음${NC}"
        return 0
    fi

    echo -e "${BLUE}정리 대상 ${#unique_list[@]}개 세션${NC}"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    for sname in "${unique_list[@]}"; do
        echo -e "  ${CYAN}${sname}${NC}"
    done
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    read -p "삭제하시겠습니까? (y/N): " c
    [[ ! "$c" =~ ^[Yy]$ ]] && echo "취소" && return 0

    local count=0
    for sname in "${unique_list[@]}"; do
        echo -e "  ${RED}삭제${NC}: ${sname}"
        "$ZELLIJ" kill-session "$sname" 2>/dev/null
        sleep 0.3
        "$ZELLIJ" delete-session "$sname" 2>/dev/null
        rm -f "/tmp/.csk-ran-${sname}" 2>/dev/null
        count=$((count+1))
    done
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo -e "${GREEN}${count}개 삭제 완료${NC}"

    # Phase 3: 끊긴 세션 추적 파일 정리
    # zellij 세션 목록에 없는데 /tmp/.csk-ran-* 가 남아있는 경우 (비정상 종료 등)
    local all_sessions stale_count=0
    all_sessions=$("$ZELLIJ" list-sessions 2>/dev/null \
        | sed $'s/\033\\[[0-9;]*m//g' \
        | awk '{print $1}')
    for flagfile in $(compgen -G "/tmp/.csk-ran-*" 2>/dev/null); do
        [[ ! -f "$flagfile" ]] && continue
        local fname="${flagfile#/tmp/.csk-ran-}"
        if ! echo "$all_sessions" | grep -qx "$fname"; then
            rm -f "$flagfile"
            echo -e "  ${YELLOW}stale flagfile 제거${NC}: ${fname}"
            stale_count=$((stale_count+1))
        fi
    done
    # 1시간 이상 된 임시 init 스크립트 정리
    local old_init
    old_init=$(find /tmp -maxdepth 1 -name ".csk-init-*.sh" -mmin +60 2>/dev/null)
    if [ -n "$old_init" ]; then
        echo "$old_init" | while IFS= read -r f; do
            rm -f "$f"
            echo -e "  ${YELLOW}old init script 제거${NC}: $(basename "$f")"
            stale_count=$((stale_count+1))
        done
    fi
    [ $stale_count -gt 0 ] && echo -e "${GREEN}추적 파일 ${stale_count}개 정리 완료${NC}"
}

# ============================================================================
# 메인
# ============================================================================

main() {
    local cmd="${1:-}"
    shift 2>/dev/null || true

    case "$cmd" in
        start)           start_all_sessions ;;
        list|ls)         list_sessions "$@" ;;
        attach|a)        attach_session "$@" ;;
        stop|kill)       stop_session "$@" ;;
        restart|r)       restart_sessions "$@" ;;
        status|s)        check_status ;;
        sync|sy)         sync_sessions ;;
        clean|cl)        clean_exited_sessions ;;
        edit|e)          edit_config ;;
        config|c)        show_config ;;
        help|--help|-h)  show_help ;;
        "")              interactive_picker ;;
        *)
            echo -e "${RED}알 수 없는 명령어: $cmd${NC}"
            echo ""
            show_help
            exit 1
            ;;
    esac
}

main "$@"
